\section{Notation, assumptions, and glossary}
\label{app:letter-A}

\subsection{Notation and standing assumptions}
\label{app:notation}

\begin{itemize}[leftmargin=1.2em,itemsep=2pt]
\item $\X$, $\mathcal{C}$, $\Y$ are the spaces of queries, contexts, and
answers; $\Y$ is finite.
\item $p_\theta(y\mid q)$ is the closed-book softmax distribution induced by
the model on query $q$.  $p_{\ctx}(y\mid q,c)$ is the context-conditioned
softmax.  Both are full-support distributions in $\Delta(\Y)$.
\item $\reli\in[0,1]$ is the latent context reliability introduced in the
setup paragraph of \Cref{sec:method}.
\item $\pi(\reli)$ is a prior on $\reli$; $p(c\mid \reli)$ is a likelihood of
the observed context.  Both are conditioned on $q$ throughout, suppressed for
brevity.
\item $S:\Delta(\Y)\!\times\!\Y\to\R$ is a strictly proper scoring rule;
log loss $S_{\log}(\hat p,y)=-\log\hat p(y)$ and Brier
$S_{\mathrm{B}}(\hat p,y)=\sum_{y'}(\hat p(y')-\one{y'=y})^2$ are the working
choices.
\item Risk $R(\hat p)=\E_{q,c,y^\star}[S(\hat p,y^\star)]$ is the standard expected score; we write $R(\hat y)$
when the policy is a 0/1 estimator with the convention $\ell(y,y^\star)=\one{y\!\neq\!y^\star}$.
\item $w^\star(q,c)\!=\!\E[\reli\mid q,c]$ is the posterior mean reliability.
When conditioning on $q$ is clear from context, we abbreviate this as
$w^\star(c)$.
\item $\KL(\mu\|\nu)$ is the KL divergence; $\TV$ is total variation;
$\JSD$ is Jensen--Shannon divergence; $H$ is Shannon entropy.
\end{itemize}

\begin{assumption}[Conditional independence]
\label{app:assumption-ci}
For all $q\in\X$, $c\in\mathcal{C}$, and $\reli\in[0,1]$,
$y^\star\!\perp\! c\mid (q,\reli)$.
\end{assumption}

\begin{assumption}[Log-linear conditional answer model]
\label{app:assumption-loglinear}
The latent reliability acts directly as the mixture weight:
\[
p(y\mid q,c,\reli)\propto p_\theta(y\mid q)^{1-\reli}p_{\ctx}(y\mid q,c)^{\reli}.
\]
The log-linear family is closed under exponentiation by $[0,1]$
since $\Y$ is finite.
\end{assumption}

\begin{assumption}[Bounded log-density ratio]
\label{app:assumption-bdd-llr}
$\sup_{y\in\Y,q\in\X,c\in\mathcal{C}}\,
\bigl|\log\frac{p_{\ctx}(y\mid q,c)}{p_\theta(y\mid q)}\bigr|\,\leq\,M<\infty$.
This is a mild full-support condition; it can be enforced in practice with
$\varepsilon$-smoothing.
\end{assumption}


\section{Related work, comparators, and theoretical antecedents}
\label{app:letter-B}

\subsection{Connection to safe Bayes and the calibration literature}
\label{app:safe-bayes-connection}

$\eta$-tempering is the inference-time analogue of safe-Bayes
shrinkage~\citep{grunwald2017,bissiri2016}, applied only to the
post-trace answer readout: a strict $\eta$-Gibbs correction that
preserves the trace's endogenous-evidence role while deflating its
overconfident influence on the final commit.  Like Platt scaling or
isotonic~\citep{guo2017oncalibration,naeini2015ece,platt1999,zadrozny2002isotonic},
it needs held-out correctness for Brier selection; unlike them, it
uses one scalar pre-argmax (so it can flip predictions, not just
recalibrate scores).  RLCR~\citep{damani2025rlcr} is the training-time
dual—shifting effective $\eta$ down via a Brier-style reward;
$\eta$-tempering is the cheaper retrofittable variant.

\section{Proof of Theorem~1 and supplementary derivations}
\label{app:letter-C}

\subsection{Proof of Theorem~\ref{thm:bayes-arbitration}}
\label{app:thm1}

\subsubsection{The Bayes action under a strictly proper scoring rule}
\label{app:thm1-bayes-action}

\begin{proposition}[Posterior predictive is the Bayes action]
\label{prop:thm1-step1}
For strictly proper $S$, the posterior predictive
$p^\star(\cdot\!\mid\!q,c)$ uniquely minimises
$\E_{y^\star\mid q,c}[S(\hat p,y^\star)]$ on $\Delta(\Y)$.
\end{proposition}

\begin{proof}
Strict propriety: $\E_{y\sim\nu}S(\nu,y)\!\le\!\E_{y\sim\nu}S(\hat p,y)$
with equality iff $\hat p\!=\!\nu$~\citep{guo2017oncalibration};
set $\nu\!=\!p^\star$ and tower over $(q,c)$.
\end{proof}

\begin{proposition}[Integral form]
\label{prop:thm1-step2}
Under \Cref{app:assumption-ci},
\begin{equation}
p^\star(y\!\mid\!q,c)\;=\;\int_0^1 p(y\!\mid\!q,c,\reli)\,p(\reli\!\mid\!q,c)\,d\reli.
\label{eq:thm1-integral}
\end{equation}
\end{proposition}

\begin{proposition}[Geometric mixture under log-linearity]
\label{prop:thm1-step3}
Under \Cref{app:assumption-loglinear},
\begin{equation}
p^\star(y\!\mid\!q,c)\;\propto\;
p_\theta(y\!\mid\!q)^{1-w^\star}\,
p_\ctx(y\!\mid\!q,c)^{w^\star},\qquad
w^\star\!=\!\E[\reli\!\mid\!q,c].
\label{eq:thm1-geommix-app}
\end{equation}
\end{proposition}

\begin{proof}[Sketch]
Log-linearity gives
$\log p(y\!\mid\!q,c,\reli)\!=\!\log p_\theta\!+\!\reli\Delta(y;q,c)\!-\!Z(\reli)$
with $\Delta\!=\!\log p_\ctx/p_\theta$.  Substituting and Taylor-expanding
$\exp(\reli\Delta)$ around $\bar r\!=\!\E[\reli\!\mid\!q,c]$ yields
$\exp(\bar r\Delta)[1\!+\!\tfrac12\sigma_r^2\Delta^2\!+\!O(\sigma_r^4)]$;
leading term renormalises to~\eqref{eq:thm1-geommix-app}; the correction
is $O(\sigma_r^2)$ and exact in the degenerate case
$\reli\!\sim\!\delta_{w^\star}$.
\end{proof}

\begin{proposition}[Variational characterization]
\label{prop:thm1-variational}
$p^\star_w\!\propto\!p_\theta^{1-w}p_\ctx^w$ uniquely minimises
$(1\!-\!w)\KL(p\|p_\theta)\!+\!w\KL(p\|p_\ctx)$ on $\Delta(\Y)$
\eqref{eq:thm1-variational}; it is the $e$-geodesic mean at parameter $w$.
\begin{equation}
p^\star_w\!=\!\arg\min_p\bigl[(1-w)\KL(p\|p_\theta)+w\KL(p\|p_\ctx)\bigr].
\label{eq:thm1-variational}
\end{equation}
\end{proposition}

\begin{proof}[Sketch]
Lagrangian on $\Delta(\Y)$ is strictly convex; stationarity gives
$\log p\!=\!(1\!-\!w)\log p_\theta\!+\!w\log p_\ctx\!-\!\text{const}$.
\end{proof}

\subsubsection{Identifiability of $w^\star$ from observable signals}
\label{app:thm1-identifiability}

Theorem~\ref{thm:bayes-arbitration} requires that the unobserved $w^\star(q,c)$
be estimable from observable signals.  We give a clean lemma.

\begin{lemma}[Reliability-signal identifiability]
\label{lem:identifiability}
Suppose $s(q,c)\in\R^d$ is a measurable signal such that $\reli\!\perp\! c\mid s(q,c)$
in distribution (i.e., $s$ is a sufficient statistic for $\reli$ given $c$).
Then $w^\star(q,c)=\E[\reli\mid q,c]=\E[\reli\mid q,s(q,c)]$ is
identifiable from any joint sample of $(s(q,c),y^\star,y_{\ctx},y_\param)$.
In particular, the plug-in
\begin{equation}
\hat w(c)\;=\;\widehat{\E}\!\left[\one{y^\star=y_{\ctx}}\,\big|\,s(q,c),\,y_{\ctx}\!\neq\!y_\param\right]
\label{eq:plugin-w}
\end{equation}
is a consistent estimator of $w^\star$ on the conflict subset.
\end{lemma}

\begin{proof}
By the tower property,
$\E[\reli\mid q,c]=\E[\E[\reli\mid q,c,s(q,c)]\mid q,c]
=\E[\reli\mid q,s(q,c)]$ since $s$ is sufficient.  The plug-in
\eqref{eq:plugin-w} is the conditional expectation of the indicator
$\one{y^\star=y_{\ctx}}$ on the conflict subset, which by definition of
$\reli$ equals $\E[\reli\mid s,y_{\ctx}\neq y_\param]$.
Standard nonparametric regression
consistency~\citep{tsybakov2009nonparametric} gives the conclusion.
\end{proof}

In practice we take $s(q,c)=(\text{retrieval score, popularity, semantic
entropy, contradiction type})$ and fit a logistic regression on
\eqref{eq:plugin-w}.  This is the implementation behind the ``Bayes proxy''
policy.

\subsubsection{Excess Bayes risk for the plug-in}
\label{app:thm1-excess-risk}

\begin{lemma}[Plug-in excess risk under log loss]
\label{lem:excess-risk-log}
Let $\hat p_w$ denote the geometric mixture with weight $w$ and let
$\hat w$ be a plug-in estimator of $w^\star$.  Under
\Cref{app:assumption-bdd-llr},
\begin{equation}
R_{\log}(\hat p_{\hat w})-R_{\log}(\hat p_{w^\star})\;\leq\;
2M^2\,\E\!\left[(\hat w-w^\star)^2\right].
\label{eq:plugin-log-risk}
\end{equation}
\end{lemma}

\begin{proof}[Sketch]
$\partial_w^2 R_{\log}\!=\!\Var_{\hat p_w}(\log p_\ctx/p_\theta)\!\le\!M^2$
under \Cref{app:assumption-bdd-llr}; Taylor at $w^\star$.
\end{proof}

\begin{lemma}[Plug-in excess risk under 0/1 loss]
\label{lem:excess-risk-zero-one}
Let $\hat y_w=\arg\max_y \hat p_w(y)$.  Then
\begin{equation}
R_{0/1}(\hat y_{\hat w})-R_{0/1}(\hat y_{w^\star})\;\leq\;
\sqrt{\E[(\hat w-w^\star)^2]}\cdot\bigl\|p_\theta-p_{\ctx}\bigr\|_1,
\label{eq:plugin-01-risk}
\end{equation}
giving the bound stated in the body's Theorem~\ref{thm:bayes-arbitration}.
\end{lemma}

\begin{proof}[Sketch]
$|R_{0/1}(\hat y_p)\!-\!R_{0/1}(\hat y_q)|\!\le\!2\TV(p,q)$;
$\TV\!\le\!\tfrac12\|\partial_w\hat p_w\|_1|\hat w\!-\!w^\star|\!\le\!\tfrac12\|p_\theta\!-\!p_\ctx\|_1|\hat w\!-\!w^\star|$;
Cauchy--Schwarz.  Rate $1/\sqrt n$ is sharp; constants are slack.
\end{proof}

\paragraph{Matching Le Cam two-point lower bound.}
The upper bound \eqref{eq:plugin-01-risk} is rate-tight at
$1/\sqrt n$.  The matching lower bound below shows no estimator
of $w^\star$ from $n$ paired (signal, outcome) observations can
beat that rate even by a constant exponent, so the plug-in is
minimax-rate-optimal.

\begin{theorem}[Le Cam two-point lower bound for $w^\star$ estimation]
\label{thm:lecam-lower}
Fix $(q,c)$ and a finite output space $\Y$ with $|\Y|\!\geq\!2$.
Suppose the contextual and parametric distributions
$p_\theta,p_{\ctx}\!\in\!\Delta(\Y)$ satisfy
$\|p_\theta-p_{\ctx}\|_1\!=\!\delta\!>\!0$.  Let $\hat w_n$ be any
(possibly randomized) estimator of $w^\star\!\in\![0,1]$ based on
$n$ iid samples
$(s_i,y_i)_{i=1}^{n}\!\sim\!\hat p_{w^\star}$.  Then
\[
\inf_{\hat w_n}\sup_{w^\star\in[0,1]}\,\E\!\left|\hat w_n - w^\star\right|
\;\geq\;\frac{1}{8\delta}\cdot\frac{1}{\sqrt{n}}.
\]
Composing with the Lipschitz dependence of $\hat p_w$ on $w$
through the geometric mixture and the standard $0/1$-loss
continuity bound, the same construction yields, for any policy
$\hat y_n$ that depends on the data only through $\hat w_n$,
\[
\inf_{\hat y_n}\sup_{w^\star\in[0,1]}\,
\bigl[R_{0/1}(\hat y_n)-R_{0/1}(\hat y_{w^\star})\bigr]
\;\geq\;\frac{\delta^2}{32}\cdot\frac{1}{\sqrt n},
\]
which matches the upper bound of \Cref{lem:excess-risk-zero-one}
up to a constant.
\end{theorem}

\begin{proof}[Sketch]
Two-point family $w_j\!=\!1/2\pm 1/(4\sqrt n)$;
$\KL(P_0^{\otimes n}\|P_1^{\otimes n})\!\le\!c_0\delta^2/4$ via
log-density linearization.  Le Cam's two-point
lemma~\citep[Thm.~2.2]{tsybakov2009nonparametric} + Markov;
Lipschitz $\|\partial_w\hat p_w\|_1\!\le\!\delta$ closes the regret.
The empirical $R^2\!=\!0.998$ at $n\!=\!2048$
(\Cref{app:oracle-diagnostics}) saturates the bound.
\end{proof}

\subsubsection{AdaCAD as a Laplace approximation}
\label{app:thm1-adacad}

\begin{lemma}[AdaCAD's JSD weight is the first-order Laplace approximation of the posterior reliability]
\label{lem:adacad-laplace}
Suppose $\reli\!\sim\!\textup{Beta}(\alpha,\beta)$ with $\alpha+\beta\to\infty$
and the likelihood $p(c\mid \reli)$ has Hellinger affinity
$1-\frac12 H^2(p_\theta,p_{\ctx})$ at $\reli=1/2$.  Then
$\E[\reli\mid c]\,=\,\tfrac12\,+\,\tfrac{1}{4}\JSD(p_\theta\|p_{\ctx})\,+\,O(\JSD^2)$
in the small-conflict regime.  Hence AdaCAD's weight
$w_{\textup{AdaCAD}}=\JSD(p_\theta\|p_{\ctx})$, after the standard $1/4$
calibration constant absorbed into the temperature, is the first-order Laplace
approximation of the posterior reliability.
\end{lemma}

\begin{proof}[Sketch]
Taylor $\log p(c\!\mid\!\reli)$ around $\reli\!=\!1/2$; symmetrising
$\KL$ under the Beta prior gives $\JSD/4$.
\end{proof}

\Cref{lem:adacad-laplace} explains why AdaCAD performs strongly empirically
without ever solving an explicit posterior: it is approximating $w^\star$ to
first order.  The quadratic remainder is what makes the full Bayes proxy
beat AdaCAD by $+0.0659$ regret on the spotlight matrix
(\Cref{tab:spotlight}).

\begin{remark}[CAD as $w=2$, CoCoA as ensemble]
CAD~\citep{shi2024cad} sets $\log\hat p\propto\log p_{\ctx}-\log p_\theta$,
i.e., $w=2$, an over-trust projection that lies outside $[0,1]$ and
deliberately over-amplifies $p_{\ctx}$.  CoCoA's three-signal score
(entropy gap, R\'enyi divergence, contextual peakedness) is a
\emph{non-Bayesian ensemble} of reliability proxies; without the posterior
update step it is consistent for $w^\star$ only when the three signals are
unbiased and equally informative, which is empirically not the case.
\end{remark}

\begin{corollary}[Existing methods are special cases]
\label{cor:special-cases}
\textup{(a)} CAD~\citep{shi2024cad} is $w\!=\!2$ (out-of-range; over-amplifies).
\textup{(b)} Always-context = $w\!=\!1$, always-parametric = $w\!=\!0$;
both are dominated by \Cref{thm:minimax}.
\textup{(c)} AdaCAD's $w\!=\!\JSD(p_\theta\|p_{\ctx})$ is the first-order
Laplace approximation of $\E[r\mid q,c]$ under a Beta$(\alpha,\beta)$ prior
with Hellinger-affinity likelihood (\Cref{lem:adacad-laplace}).
\textup{(d)} CoCoA's three-signal heuristic is a non-Bayesian ensemble of
proxies for $\E[r\mid q,c]$ and need not converge to $w^\star$ without
additional calibration assumptions.
\end{corollary}

\subsubsection{Sufficient signal statistic for log-linear arbitration}
\label{app:thm1-strengthenings}

Define $\mathcal{F}_{\mathrm{ll}}\!=\!\{\hat p_\phi\!:\!\phi\!:\!\mathcal{S}\!\to\![0,1]\}$
with $\hat p_\phi\!\propto\!p_\theta^{1-\phi(s)}p_\ctx^{\phi(s)}$ and
local Fisher info $V(q,c)\!=\!\partial^2_w R|_{w^\star}\!>\!0$.

\begin{theorem}[Sufficient signal statistic]
\label{thm:t1-sufficient}
Under T1's assumptions, on log loss,
\begin{equation}
R(\phi)-R^\star\;=\;\tfrac12\E_s[\bar V(s)(\phi(s)-\phi^\star(s))^2]\;+\;O(\|\phi-\phi^\star\|_{L^3}^3),
\label{eq:t1-suff-quadratic}
\end{equation}
$\phi^\star(s)\!=\!\E[\reli\!\mid\!s]$, $\bar V\!=\!\E[V\!\mid\!s]$.
$\phi^\star$ is the unique minimizer of $R$ on $\mathcal{F}_{\mathrm{ll}}$
(a.s.\ in $s$).
\end{theorem}

\begin{proof}[Sketch]
Per-$(q,c)$ Taylor: $F(w;q,c)\!=\!F^\star\!+\!\tfrac{V}{2}(w-w^\star)^2\!+\!O(\cdot)$
since $w^\star\!=\!\E[\reli\!\mid\!q,c]$ is stationary
(Prop.~\ref{prop:thm1-variational}).  Tower-condition on $s$ and
minimise the quadratic in $\phi(s)$: weighted projection
$\phi^\star(s)\!=\!\E[V\E[\reli\!\mid\!q,c]\!\mid\!s]/\E[V\!\mid\!s]\!=\!\E[\reli\!\mid\!s]$
when $V$ is $s$-measurable / $\reli$-independent given $s$.
\end{proof}

\begin{corollary}[Decision-theoretic sufficiency]
\label{cor:t1-sufficient}
$\E[\reli\!\mid\!s]$ is Blackwell-sufficient for log-linear
arbitration: $T(s)$ is risk-optimal iff $\E[\reli\!\mid\!s]$ is
$\sigma(T)$-measurable.
\end{corollary}

\begin{corollary}[Coarsening penalty]
\label{cor:t1-coarsening}
$\phi_T^\star(T)\!:=\!\E[\bar V\phi^\star\!\mid\!T]/\E[\bar V\!\mid\!T]$
gives $R(\phi_T^\star)\!-\!R(\phi^\star)\!=\!\tfrac12\E[\bar V(\phi_T^\star\!-\!\phi^\star)^2]\!+\!O(\cdot)$;
$\bar V\!\ge\!v_{\min}\!>\!0$ replaces $\bar V$ by $v_{\min}$ in
the lower bound, equality (cubic-up-to) iff $\phi^\star$ is
$\sigma(T)$-measurable.
\end{corollary}

CAD/AdaCAD/CoCoA approximate $\E[\reli\!\mid\!s]$ at increasing
accuracy (\Cref{tab:special-cases-residual}); residual gap tracks
how well each spans the sufficient statistic.

\subsubsection{Sequence-level extension: Corollary~\ref{cor:sequence-extension} and proof}
\label{app:t1-sequence-extension}

\begin{corollary}[Sequence-level extension]
\label{cor:sequence-extension}
On $\Y\!=\!\mathcal{V}^{\le T_{\max}}$, the T1 variational
characterization holds verbatim; per-token chain mixture
$p_w^{\mathrm{tok}}\!\propto\!\prod_t p_\theta^{1-w}p_\ctx^w$ equals
the sequence-level mixture exactly when the chain partition
factorises, otherwise differs by partition-mismatch KL.
\end{corollary}

\begin{lemma}[Chain-vs-sequence gap]
\label{lem:chain-mixture-gap}
$\KL(p_w^{\mathrm{tok}}\|p_w^\star)\!=\!\sum_t\E_{y_{<t}\sim p_w^{\mathrm{tok}}}\log Z_t(y_{<t})\!-\!\log Z_w$,
zero iff per-step partitions chain-decompose.
\end{lemma}

Gap is zero on discrete-candidate benchmarks; bounded by
$T\log|\mathcal{V}|$ on free-form.  CAD/AdaCAD/CoCoA are special
cases in both regimes.

\subsubsection{Boundary behaviour and signal robustness}
\label{app:thm1-boundary}

The geometric mixture in \Cref{thm:bayes-arbitration} is well behaved
on the operating regime that LLM softmax outputs actually occupy, but
two boundary cases and one robustness question deserve explicit
treatment.

\paragraph{Zero-mass support mismatch} does not arise (LLM softmax
has full vocabulary support); a $\epsilon\!\ge\!10^{-30}$ floor
costs $\sim\!10^{-25}$ nat, below other approximation gaps.
\textbf{Argmax disagreement} (mode flip at one $w\!\in\!(0,1)$) is
the per-cell signal $\eta$-tempering and $w^\star$ arbitrate.
\textbf{Adversarial signal} $\|\delta\|_2\!\le\!B$ inflates excess
risk by $\le(\|\hat\beta\|_2 B/4 + \|\hat w\!-\!w^\star\|_{L^2})\|p_\theta\!-\!p_\ctx\|_1$;
empirically $\|\hat\beta\|_2\!\in\![2.31,2.96]$, $B\!=\!0.1$
gives worst-case inflation $\sim\!0.074$.  \textbf{Conjugacy} is
not required: the geometric mixture is the optimal log-linear
posterior projection regardless of $\pi$'s shape, with
$O(\sigma_r^2)$ correction (\Cref{app:t1-variance-correction})
the only place $\pi$ enters.

\subsection{Theorem-1 variance correction (empirical)}
\label{app:t1-variance-correction}

Empirical $\KL(p^\star\|\hat p_{w^\star})$ on the spotlight matrix

\subsection{Detailed proof of the variational characterization}
\label{app:variational-detail}

\textbf{Lagrangian.}  $\Lambda_w(p)\!=\!(1\!-\!w)\KL(p\|p_\theta)\!+\!w\KL(p\|p_\ctx)$
is jointly convex~\citep[Thm.~2.7.2]{cover2006elements}.
Stationarity gives
$\log p\!=\!(1\!-\!w)\log p_\theta\!+\!w\log p_\ctx\!-\!\text{const}$,
i.e., $p\!\propto\!p_\theta^{1-w}p_\ctx^w$; strict convexity
yields uniqueness.
\textbf{$e$-geodesic.}  In exponential-family $e$-coordinates,
$p_w^*$ is exactly the $e$-geodesic between $p_\theta$ and $p_\ctx$
at parameter $w$.
\textbf{Brier equivalence.}  Brier
$\E_{y^\star\sim p^\star}S_B(\hat p,y^\star)\!=\!\|\hat p\!-\!p^\star\|_2^2\!+\!\text{const}$
is minimised at $\hat p\!=\!p^\star$, identifying the log-loss
minimiser by \Cref{prop:thm1-step1,prop:brier-log-equivalence}.


\subsection{Proof of the axiomatic uniqueness of the geometric mixture}
\label{app:thm1-axiomatic}

\begin{proof}[Proof of \Cref{thm:t1-axiomatic}]
Define $\phi(a,b,w)\!:=\!\Phi(p,q,w)(y)/\Phi(p,q,w)(y')$ with
$a\!=\!p(y)/p(y')$, $b\!=\!q(y)/q(y')$.  (A4) makes $\phi$
depend on $(a,b,w)$ alone; (A1)+(A2)+(A3) give boundary,
idempotence, and label-equivariance constraints.  Chaining
ratios across three outcomes yields the multiplicative Cauchy
equation
$\phi(a_1a_2,b_1b_2,w)\!=\!\phi(a_1,b_1,w)\phi(a_2,b_2,w)$;
continuity gives the unique solution
$\phi(a,b,w)\!=\!a^{\alpha(w)}b^{\beta(w)}$.  Boundary
$\phi(a,b,0)\!=\!a$, $\phi(a,b,1)\!=\!b$ and idempotence
$\alpha(w)\!+\!\beta(w)\!=\!1$ pin $\alpha\!=\!1\!-\!w$,
$\beta\!=\!w$, so $\Phi(p,q,w)\!\propto\!p^{1-w}q^{w}$.
\end{proof}

\begin{remark}[Axiom roles]
(A1) rules out trivial combiners; (A2) fixes $w$ as a mixing
weight; (A3) is label-equivariance; (A4) (LLR-invariance) is the
substantive axiom: dropping it admits the linear pool $(1\!-\!w)p\!+\!wq$.
\end{remark}

\begin{lemma}[EB excludes reliability-adaptive pools]
\label{lem:eb-excludes-adaptive}
Under External Bayesianity~\citep[Thm.~3.1]{genest1986characterization},
the unique pool is $\Phi(p,q)\!\propto\!p^{1-w_0}q^{w_0}$ with
\emph{constant} $w_0$.  Any signal-conditioned $w(s)$ pool
(AdaCAD, CoCoA) or constant outside $[0,1]$ (CAD, $w\!=\!2$)
violates EB but satisfies T1.
\end{lemma}

\begin{proof}[Sketch]
EB's commutation forces $w$ to depend only on the support, not on
external signals; non-constant $w(s)$ depends on $s$ invisible to
Bayes updates on $(p,q)$.
\end{proof}


\subsection{Proof of the information-theoretic minimax floor (T2+)}
\label{app:thm2-info}

\begin{proof}[Proof of \Cref{thm:t2-info}]
Fix non-adaptive $\hat y$ depending on $(q,c)$ only through
$s\!=\!s(q,c)$ ($I(r;\hat y\mid s)\!=\!0$).  Strict properness +
convex Bayes risk $\mathcal{R}$ give the regret decomposition
$R(\hat y)\!-\!R(\hat y^\star)\!\ge\!\E[\mathcal{R}(\hat y(q,c))\!-\!\E_r\mathcal{R}(\hat y^\star(q,c,r))]$.
Strong convexity ($\mu\!\ge\!1/(L|\Y|)$ for log loss; $\mu\!=\!2$
Brier) plus the conditional-mean optimization gives
$\Var_{r\mid q,c}[\hat y^\star]\!\ge\!\kappa^2\sigma_r^2(q,c)$
(Lipschitz in $r$).  Pinsker
$\mathrm{TV}^2\!\le\!\tfrac12\KL$ + Cauchy--Schwarz
$\|\cdot\|_2^2\!\ge\!4\mathrm{TV}^2/(|\Y|\!-\!1)$ collect to
$\ge\tfrac{1}{8(|\Y|-1)^2}\sigma_r^2(q,c)$; tower over $s$.
\end{proof}

\begin{remark}
Plug-in matches at $\Theta(\E[\sigma_r^2(s)])\!+\!O(n^{-1/2})$;
two-point reduction is tight, $|\Y|\!-\!1$ slack is the
$\ell_2\!\to\!\ell_1$ conversion cost; Brier replaces the constant
with $1/(2(|\Y|\!-\!1)^2)$.
\end{remark}


\subsection{Proof of the closed-form Brier improvement (T3+)}
\label{app:thm3-intervention}

\begin{proof}[Proof of \Cref{thm:t3-intervention}]
$\hat p_\eta\!\propto\!p_K^\eta$.  Taylor-expand
$\mathcal{B}(\eta)$ around $\eta^\star$: optimality kills the
first-order term; second derivative identity
$\mathcal{B}''(\eta^\star)\!=\!2\Var_{p_K}(\bm{1}\{Y\!=\!y^\star\})\!+\!o(1)\!=\!2V_{\mathrm{post}}\!+\!o(1)$.
Combining,
$\Delta\mathrm{Brier}\!=\!(1\!-\!\eta^\star)^2 V_{\mathrm{post}}\!+\!O((1\!-\!\eta^\star)^3)$,
strictly positive when $\eta^\star\!<\!1$ and
$V_{\mathrm{post}}\!>\!0$.  No stable-answer condition.
\end{proof}

\subsubsection{Closed form for $\eta^\star$ on binary-collapse cells}
\label{app:eta-closed-form}

\begin{lemma}[Binary-collapse $\eta^\star$]
\label{lem:eta-closed-form-binary}
Suppose $p_K$ and the Bayes-optimal $q^\star$ both concentrate on
two answers with masses $\{a,1\!-\!a\}$ and $\{b,1\!-\!b\}$
respectively (binary-collapse).  Then the Brier-optimal tempering
$\eta^\star$ on this cell satisfies
$\eta^\star\!=\!\mathrm{logit}(b)/\mathrm{logit}(a)$, and
linearizing at the Berk--Nash equilibrium of
\Cref{thm:berk-nash-rate} gives
$\eta^\star\!=\!1\!-\!D^\star/(\bar\eta\cdot\mathrm{logit}(b))\!+\!O(D^{\star\,2})$.
\end{lemma}

\begin{proof}[Sketch]
On two atoms, $\mathrm{logit}\,\hat p_\eta\!=\!\eta\,\mathrm{logit}(a)$;
Brier-optimal $\hat p\!=\!q^\star$ closes
$\eta^\star\!=\!\mathrm{logit}(b)/\mathrm{logit}(a)$.  Linearisation
uses Berk--Nash $b\!=\!a\,e^{-D^\star/\bar\eta}+O(D^{\star 2})$
(\Cref{prop:berk-nash-reduction}).
\end{proof}

This is a one-line algebra fact; we surface it because it lets
practitioners skip cross-validation on $\sim\!42\%$ of conflict
cells (\Cref{tab:eta-closed-form-empirical}) at the cost of
$\le\!0.004$ Brier vs.\ $5$-fold CV.  The result is small but
useful; it is not load-bearing for the paper's claims.

\begin{remark}[Relation to the safe-rate result]
\Cref{thm:eta-bayes} (\Cref{cor:two-regime}) characterizes
\emph{when} $\eta^\star\!<\!1$ on the stable-answer subset:
overshoot of $\bar\eta$ creates the non-decreasing calibration
gap, which forces $\eta^\star\!<\!1$ for Brier-optimal tempering.
\Cref{thm:t3-intervention} then quantifies the resulting Brier
improvement.  The two theorems are complementary: the safe-rate
result says \emph{when} $\eta$-tempering should help; the
intervention result says \emph{by how much} when it does, with no
subset condition required.
\end{remark}


\subsection{Detailed proof of Pinsker-style excess risk}
\label{app:pinsker-detail}

The geometric mixture is a one-parameter exponential family with
$T(y)\!=\!\log p_\ctx/p_\theta$ and Fisher
$I(w)\!=\!\Var_{p_w}T\!\le\!M^2/4$ (Popoviciu under
\Cref{app:assumption-bdd-llr}).  Integrating
$\KL(p_w\|p_{w'})\!\le\!\tfrac{M^2}{8}(w\!-\!w')^2$ + Pinsker
gives $\TV\!\le\!\tfrac{M}{4}|w\!-\!w'|$.  Combined with the 0/1
continuity bound $|R_{0/1}(\hat y_p)\!-\!R_{0/1}(\hat y_q)|\!\le\!2\TV$
and Cauchy--Schwarz,
\begin{equation}
R_{0/1}(\hat y_{\hat w})-R_{0/1}(\hat y_{w^\star})\;\le\;\tfrac{M}{2}\|\hat w-w^\star\|_{L^2},
\label{eq:tight-excess-risk}
\end{equation}
order-optimal vs.\ Le Cam (\Cref{thm:lecam-lower}) up to constant $2$.

\subsection{If conditional independence or log-linearity fails}
\label{app:ci-failure}\label{app:loglin-failure}

\textbf{CI failure} ($y^\star$ depends on $c$ given $(q,r)$): the
geometric mixture is no longer the unique Bayes-optimal predictive;
the integral form
$p^\star(y\!\mid\!q,c)\!=\!\int p(y\!\mid\!q,c,r)p(r\!\mid\!q,c)\,dr$
is.  We see no evidence of CI failure on our cells (per-cell
residual $y^\star\!-\!\hat y_{w^\star}$ has no systematic
$c$-dependence beyond $r$); a held-out
$I(y^\star;c\!\mid\!q,r)$ test would diagnose pathological cases.

\textbf{Log-linearity failure} (e.g., a hard gate at threshold $\tau$):
the gated alternative
$p^\star\!=\!p_\ctx\!\one{w^\star\!\ge\!\tau}\!+\!p_\theta\!\one{w^\star\!<\!\tau}$
is dominated by the soft mixture in the small-conflict regime and
approximately equal in the large-conflict regime; empirically,
$+0.012$ regret worse on the spotlight matrix.


\subsection{Extended discussion of the AdaCAD Laplace approximation}
\label{app:adacad-extended}

\textbf{Second-order correction.}
$\E[\reli\!\mid\!c]\!=\!\tfrac12\!+\!\tfrac{\JSD}{4}\!+\!\tfrac{\JSD\cdot[\JSD-\Var_{\textrm{Beta}}r]}{16}\!+\!O(\JSD^3)$;
correction $0.005$--$0.020$ on the spotlight matrix matches the
$+0.066$ AdaCAD-vs-Bayes regret gap after weighting by
$\|p_\theta\!-\!p_\ctx\|_1$.
\textbf{CoCoA.}  The three-signal heuristic does not Laplace-identify
because the signals are correlated.  Re-fitting an L2-regularised
combiner on our calibration recovers $-0.142$ regret (CoCoA itself
$-0.128$; Bayes $-0.172$).
\textbf{AdaCAD failure mode.}  Trails Bayes most where $\JSD$ is
large (ConflictBank-conflict: $+0.11$ at JSD $\sim\!1.8$ nat;
PopQA: $+0.04$ at JSD $\sim\!0.4$).


\subsection{Extended discussion of conditional independence}
\label{app:ci-extended}

CI ($y^\star\!\perp\!c\!\mid\!q,r$): ``$c$ is sufficient for
$r$.''  Exactly true in synthetic oracle; approximately true when
conflict mechanism is mediated by a clean reliability variable.
Breaks if $c$ contains side-channel info (verbatim answer, format
cue); diagnostic in App.~\ref{app:ci-failure} detects.  Without
CI, the predictive
$p^\star(y\!\mid\!q,c)\!=\!\int p(y\!\mid\!q,c,r)p(r\!\mid\!q,c)dr$
is still the right object but loses its closed form.


\subsection{Supplementary derivation, Pinsker for log-loss}
\label{app:pinsker-logloss}

This section expands the second-order argument behind
\Cref{lem:excess-risk-log}.

\begin{proposition}[Second-order log-loss sensitivity along the geometric path]
\label{prop:logloss-second-order}
Let $p_w\propto p_\theta^{1-w}p_{\ctx}^w$ and define
$R_{\log}(w):=-\E_{y^\star}[\log p_w(y^\star)]$.  Under
\Cref{app:assumption-bdd-llr}, $R_{\log}$ is twice continuously
differentiable and satisfies
\[
R_{\log}''(w)\;\leq\;4M^2\qquad\text{for all }w\in[0,1].
\]
Consequently, if $w^\star$ minimizes $R_{\log}$ on $[0,1]$, then
\[
R_{\log}(w)-R_{\log}(w^\star)\;\leq\;2M^2(w-w^\star)^2,
\]
which is the pointwise version of \Cref{lem:excess-risk-log}.
\end{proposition}

\begin{proof}
Write
$\partial_w^2 R_{\log}\!=\!\Var_{p_w}(\log p_\ctx/p_\theta)\!\le\!(2M)^2$
under \Cref{app:assumption-bdd-llr}; Taylor at $w^\star$ closes.
\end{proof}

\subsubsection{Plug-in upper bound for adaptive policies}

\section{Proof of Theorem~3, assumption audits, and multilingual evidence}
\label{app:letter-E}

\subsection{Proof of Theorem~\ref{thm:eta-bayes}}
\label{app:thm3}

\subsubsection{Setup and the $\eta$-generalized posterior}
\label{app:thm3-setup}

Bissiri--Holmes--Walker~\citep{bissiri2016} update on family
$\mathcal{M}\!=\!\{q_\xi\}$:
\begin{equation}
\pi_\eta(\xi\!\mid\!z)\!\propto\!\pi(\xi)\exp(-\eta\ell(z;\xi)),\quad
\ell\!=\!-\log q_\xi(z).
\label{eq:gen-bayes-app}
\end{equation}
For misspecified $\mathcal{M}$, \citet{grunwald2017} give a safe
rate $\bar\eta\!\in\!(0,1]$ above which the posterior
hypercompresses.  In the LLM CoT setting, $z_t\!\sim\!p_t$ from
the model itself (endogenous evidence;~\citealp{esponda2016,fudenberg2021}).

\subsubsection{The safe-rate lemma}
\label{app:thm3-shrinks}

\begin{lemma}[Larger $D^\star$ shrinks the safe rate]
\label{lem:eta-bar-monotone}
$\bar\eta\!\le\!c_{|\Y|,\pi}/D^\star$ where
$D^\star\!=\!\KL(\PP^\star\|q_{\xi^\star})$ is the KL projection
distance and $c_{|\Y|,\pi}$ collects the bracketing-entropy and
prior dependence~\citep[Thm.~1.6]{grunwald2017}.  On finite $\Y$,
$\log N_{[]}(\varepsilon,\Delta(\Y))\!\le\!(|\Y|\!-\!1)\log(3/\varepsilon)$
makes the entropy integral finite; the paper uses only the
monotonicity $\bar\eta\!\propto\!1/D^\star$.
\label{app:thm3-shrinks-explicit}
\end{lemma}

\subsubsection{Reduction to Berk--Nash dynamics: proof of Proposition~\ref{prop:berk-nash-reduction}}
\label{app:berk-nash-reduction}

The reduction needs only continuity of the expected-update operator
on $\Delta(\Y)$, not parametric $\mathcal{M}$.

\begin{definition}[Auxiliary Berk--Nash game $G_{q,c,\eta}$]
\label{def:bn-game}
Fix $(q,c)$ and a rate $\eta\in(0,1]$.  The auxiliary game has a
single agent with belief space $\Delta(\Y)$, action space identified
with the belief itself (the model's autoregressive distribution
\emph{is} its action), and a data-generating distribution
$\mu(\cdot\mid p,q,c)$ that depends on the agent's belief through
the sampling kernel of \eqref{eq:gen-bayes}.  The agent's
best-response operator is the $\eta$-Bayes update
\[
B_\eta(p)(y)\;=\;\frac{p(y)\,\E_{z\sim\mu(\cdot\mid p,q,c)}\!\exp(-\eta\,\ell(y;q,c,z))}
{\sum_{y'} p(y')\,\E_{z\sim\mu(\cdot\mid p,q,c)}\!\exp(-\eta\,\ell(y';q,c,z))}.
\]
A \emph{Berk--Nash equilibrium} of $G_{q,c,\eta}$ is a fixed point
$p^\star=B_\eta(p^\star)$.
\end{definition}

\begin{lemma}[Continuity and existence]
\label{lem:phi-continuous}
With $\ell\!\in\![0,L]$ and $\mu(\cdot\mid p,q,c)$ TV-Lipschitz in
$p$, $B_\eta$ is continuous on $\Delta(\Y)$ and has a fixed point
(Brouwer).
\end{lemma}

\begin{proof}
$\E_z\exp(-\eta\ell)\!\in\![e^{-\eta L},1]$ is continuous in $p$ via
dominated convergence; ratio of continuous, lower-bounded terms is
continuous; $\Delta(\Y)$ compact convex.
\end{proof}

\begin{lemma}[Limit of tracked trajectories]
\label{lem:bn-limit}
If $p_t\!\to\!p_\infty$ and $\|p_{t+1}\!-\!B_\eta(p_t)\|_{TV}\!\to\!0$
under \Cref{lem:phi-continuous}, then $p_\infty\!=\!B_\eta(p_\infty)$.
\end{lemma}

\begin{proof}
Triangle inequality with continuity of $B_\eta$.
\end{proof}

\begin{proposition}[$\Phi_\eta$ fixed points $=$ Berk--Nash equilibria]
\label{prop:berk-nash-reduction}
Under \Cref{lem:phi-continuous}, $\Phi_\eta\!=\!B_\eta$ fixed points
are exactly Berk--Nash equilibria of $G_{q,c,\eta}$; any tracked
limit (\Cref{lem:bn-limit}) is one of them.
\end{proposition}

\begin{proof}
Brouwer (\Cref{lem:phi-continuous}) plus the definition give the
identification; \Cref{lem:bn-limit} closes the trajectory case.
\end{proof}

\paragraph{Trajectory bridge.}
\label{app:thm3-tracking-honest}
Realized $p_{t+1}(y)\!\propto\!p_t(y)\exp(-\eta\,\ell(y;q,c,z_t))$,
$z_t\!\sim\!\mu(\cdot\!\mid\!p_t,q,c)$; expected operator
$B_\eta(p_t)(y)\!\propto\!p_t(y)\E_z\exp(-\eta\ell(y;q,c,z))$.

\begin{theorem}[Tracking and unconditional Berk--Nash convergence]
\label{thm:tracking-tau0}\label{cor:bn-convergence-tau0}\label{thm:tracking-tau-pos}
\textup{(i)~$\tau\!=\!0$, greedy $\mu\!=\!\delta_{z^\star(p)}$:}
$\|p_{t+1}\!-\!B_\eta(p_t)\|_{\TV}\!=\!0$ exactly, so $\{p_t\}$ is the
deterministic continuous iteration of $B_\eta$; subsequential limits
are Berk--Nash fixed points unconditionally, and the safe-rate
shrinkage of \Cref{thm:eta-bayes} holds on the realized trajectory
(stable-answer subset of \Cref{lem:eta-overshoot}).
\textup{(ii)~$\tau\!>\!0$, $\ell\!\in\![0,L]$, stable-answer
$p_t(y^\star)\!\ge\!1\!-\!\rho$ ($\rho\!<\!1/2$), per-step kernel
variance $V_t\!:=\!\sup_y\Var_{z\sim\mu}\ell$:}
\begin{equation}
\E\|p_{t+1}\!-\!B_\eta(p_t)\|_{\TV}^{2}
\;\le\;\frac{(\eta L)^2 V_t}{(1-\rho)^2},
\label{eq:tracking-tau-pos}
\end{equation}
with matching $\sqrt{V_t/\delta}$ high-prob.\ bound.  $L^2$ tracking
gives Berk--Nash convergence in expected $\TV$ on subsequences
where the kernel concentrates.
\end{theorem}

\begin{proof}[Sketch]
(i) $\delta_{z^\star}$ collapses the expectation to a point;
numerators/normalizers agree exactly.
(ii) $g_t(y,z)\!=\!\exp(-\eta\ell)$, $|g_t'|\!\le\!\eta$, so
$\Var_z g_t\!\le\!(\eta L)^2 V_t$ (MVT); Cauchy--Schwarz with
$Z_t^{\mathrm{exp}}\!\ge\!(1\!-\!\rho)e^{-\eta L}$; Markov.
\end{proof}

Empirically: median $\TV$ discrepancy $0.041$ at $K\!=\!1024$ on
stable-answer (90th pct $0.118$), consistent with $\sqrt{V_t}$.
Almost-sure pathwise tracking at $\tau\!>\!0$ is open and not
required by $\eta$-tempering in practice.

\paragraph{On uniqueness of the Berk--Nash equilibrium.}
$\Phi_\eta$ has a fixed point by Brouwer (\Cref{lem:phi-continuous});
uniqueness is not assumed.  In our setting two refinements rule out
spurious equilibria empirically.  First, on the stable-answer subset
the linearized iteration is hyperbolic
($\widehat\rho^\star\!<\!1$ on every conflict cell of
\Cref{tab:berk-nash-rate-empirical}), so trajectory limits selected
by gen-Bayes are isolated.  Second, on cells where the trace
concentrates around a single answer mode ($81\%$ of wrong-answer
cells, median margin $8.5$ logits), the Berk--Nash basin is a
neighborhood of the answer-mode delta; competing equilibria require
the trace kernel to bifurcate, which we do not observe in any logged
trajectory.  When the trace is multi-modal (top-2 mass within $0.1$),
the analysis reverts to the conditional safe-rate inequality on the
stable-answer subset.

\paragraph{On the $\widehat\eta_{\mathrm{eff}}$ proxy.}
$\widehat\eta_{\mathrm{eff}}$ is the trace-token logit slope at the
answer mode (\Cref{app:thm3}) and tracks injected $\eta$ on the
synthetic grid with Spearman $1.0$ and $\le\!2.03\%$ relative error
(\Cref{tab:eta-proxy-id}).  We use it as a relative-rate diagnostic,
not as point identification of the latent rate on real traces;
slice-level $\eta^\star$ selection by held-out Brier
(\Cref{tab:eta-per-slice}) does the deployment work.

\paragraph{Empirical check of the conditional-independence assumption.}
\Cref{app:assumption-ci} requires
$y^\star\!\perp\!c\mid(q,r)$.  On a held-out
ConflictBank fold ($n\!=\!500$) we estimate the conditional mutual
information $I(y^\star;c\mid q,\hat r)$ via the
$k$NN-based estimator of~\citet{kraskov2004estimating}; pooled
$\widehat I\!\le\!0.018$ nat across cells, with $99\%$
permutation-bootstrap intervals straddling zero on $24/25$ cells
(only PopQA-low-popularity is borderline at $\widehat I\!=\!0.041$,
$95\%$ CI $[0.003,0.082]$).  CI is therefore a working approximation,
with PopQA-low-popularity flagged as the one slice where a residual
$c$-channel is detectable.

\paragraph{Empirical check of the log-linearity assumption.}
The synthetic sweep (App.~\ref{app:loglin-robustness}) bounds the
worst-case $R^2$ under deliberate violation.  On real logits we run
a Hessian-spectrum probe: at the per-cell argmin of $R(w)$ we
estimate $\partial^2_w R$ via finite differences and compare to the
log-linear-family prediction $\Var_{p_w}(\log p_\ctx/p_\theta)$.
Pooled relative error across the 25 spotlight cells is $\le\!7\%$
(median $3.1\%$, max $12.4\%$ on Llama-70B/PopQA), well below the
non-trivial-violation threshold ($\rho\!=\!0.6$,
App.~\ref{app:loglin-robustness}).  Real LLM logits sit in the
``approximate log-linear'' regime where the geometric mixture is the
right plug-in target.

\subsubsection{Berk--Nash equilibrium of the gen-Bayes update (legacy lemma)}
\label{app:thm3-berk-nash}

\begin{lemma}[Convergence to a Berk--Nash equilibrium]
\label{lem:berk-nash}
Under the conditions of Proposition~\ref{prop:berk-nash-reduction},
if a decoding trajectory $\{p_t\}_{t\geq 1}$ converges and satisfies
$\|p_{t+1}-B_\eta(p_t)\|_{TV}\to 0$, then its limit is a fixed point of
$B_\eta$, equivalently a Berk--Nash equilibrium of $G_{q,c,\eta}$.
\end{lemma}

\begin{proof}
Immediate from \Cref{lem:bn-limit} under the stated tracking
hypothesis.
\end{proof}

\subsubsection{Confidence sharpening can outpace accuracy}
\label{app:thm3-sharpening}

\begin{lemma}[$\eta>\bar\eta$ implies non-decreasing calibration gap]
\label{lem:eta-overshoot}
Suppose $\eta>\bar\eta$ in the iterated update~\eqref{eq:gen-bayes-app}.
Then for any $\hat y_t$ that is the mode of $p_t$,
\begin{equation}
G(t):=\E\!\left|\E_{p_t}\one{Y\!=\!y^\star}-\one{\hat y_t = y^\star}\right|
\quad\textup{is non-decreasing in $t$ on the stable-answer subset.}
\nonumber
\end{equation}
\end{lemma}

\begin{proof}[Sketch]
On the stable-answer subset $\hat y_t$ is constant, so accuracy in
$G(t)$ is fixed; $\eta\!>\!\bar\eta$ sharpens around the
misspecified projection, raising confidence without raising
accuracy.
\end{proof}

\begin{proposition}[Stable/flip decomposition]
\label{prop:t3-decomposition}
For $S_{t_0:t_1}\!=\!\{\hat y_{t_0}\!=\!\hat y_{t_1}\}$ and
$g_t\!=\!|\E_{p_t}\one{Y\!=\!y^\star}\!-\!\one{\hat y_t\!=\!y^\star}|\!\in\![0,1]$,
$\Delta G\!\ge\!\PP(S)\Delta G^S\!-\!\PP(F)$ via the law of total
expectation and $\Delta G^F\!\ge\!-1$.
\end{proposition}

\begin{corollary}[Whole-population inheritance]
\label{cor:t3-decomposition}
If $\eta\!>\!\bar\eta$ and $\Delta G^S\!\ge\!\gamma$, with flip
probability $\PP(F)\!\le\!\delta$, then
$\Delta G\!\ge\!(1\!-\!\delta)\gamma\!-\!\delta$.
\end{corollary}

\subsubsection{Putting it together}
\label{app:thm3-together}

\begin{proof}[Proof of Theorem~\ref{thm:eta-bayes}]
Under the theorem's standing assumption that conflict increases the
projection distance $D^\star(q,c)$ relative to matched controls,
\Cref{lem:eta-bar-monotone} implies a smaller safe rate on conflict
cells.  Under the tracking hypothesis of \Cref{lem:berk-nash}, the
iterated CoT update converges to a Berk--Nash equilibrium that is
generically not the truth.  By
\Cref{lem:eta-overshoot}, when $\eta>\bar\eta$ the calibration gap $G(t)$ is
non-decreasing on the stable-answer subset; this is the
saturation regime of Corollary~\ref{cor:two-regime}.  When $\eta<\bar\eta$,
the standard \citet{kleijn2006} contraction holds and the predictive accuracy
catches up with confidence, giving the peak-then-recover regime.

For RLVR-trained models, verifiable-reward RL (GRPO) sharpens token-level
logits toward winning trajectories, increasing the effective $\eta$ in
\eqref{eq:gen-bayes-app}; \citet{bereket2025grpo} document this as
GRPO-induced overconfidence in binary forecasting.  Hence
$\eta_{\mathrm{RLVR}}>\eta_{\mathrm{RLHF}}$ at matched scale.  In a conflict
regime where $\eta_{\mathrm{RLVR}}>\bar\eta(q,c)>\eta_{\mathrm{RLHF}}$, RLVR
saturates while RLHF does not.  This is the training-type-conditional
saturation predicted by the theorem and observed in \Cref{tab:theorem3-matrix-full}.
\end{proof}

\begin{corollary}[Two-regime calibration trajectory]
\label{cor:two-regime}
The trajectory $G(K)$ has two qualitative shapes. \textup{(a)}
\emph{Peak-then-recover} for $\eta\!\lesssim\!\bar\eta$. \textup{(b)}
\emph{Saturation} for $\eta\!>\!\bar\eta$, on the stable-answer
subset.  The transition is benchmark-dependent because $\bar\eta(q,c)$
depends on $\KL(p_\theta\|p_{\ctx})$.  Empirically, the same
benchmark splits the same model between the two regimes
(\Cref{tab:theorem3-matrix-full}, \Cref{fig:theorem3-curves}); the
benchmark dependence is a direct consequence of the safe-rate lemma
above.
\end{corollary}

\subsection{Local Berk--Nash convergence rate, proof of \Cref{thm:berk-nash-rate} and empirical estimation}
\label{app:berk-nash-rate}

\paragraph{Setup and proof.}
Tangent space $T_{p^\star}\!=\!\{u\!:\!\sum_y u(y)\!=\!0\}$ with
$\|\cdot\|_2$.  Write $p_t\!=\!p^\star\!+\!\epsilon_t$.  Taylor at
$p^\star\!=\!\Phi_\eta(p^\star)$,
\begin{equation}
\epsilon_{t+1}\;=\;J\,\epsilon_t\;+\;R(\epsilon_t),
\qquad J\!:=\!D\Phi_\eta(p^\star),\qquad
\|R(\epsilon)\|_2\!\le\!M\|\epsilon\|_2^{\,2}
\label{eq:bn-linearization}
\end{equation}
where $M$ is the $C^2$-norm bound.  Gelfand: for any
$\rho\!\in\!(\rho^\star,1)$, $\|J^n\|\!\le\!C_J(\rho)\rho^{\,n}$.
Choose $\delta\!\le\!(\rho\!-\!\rho^\star)/(2MC_J(\rho))$, $T_0$ s.t.\
$C_J(\rho)\rho^{T_0}\|\epsilon_0\|\!\le\!\delta$; induction gives
$\|\epsilon_t\|_2\!\le\!C(p_0,\rho)\rho^{\,t}$, and
$\TV\!\le\!\tfrac{\sqrt{|\Y|}}{2}\|\cdot\|_2$ folds in.  This proves
\eqref{eq:bn-rate}.

\begin{corollary}[Non-asymptotic $T_0$ from observable quantities]
\label{cor:T0-nonasymptotic}
Fix $\rho\!\in\!(\rho^\star,1)$.  The threshold time $T_0$ in
\Cref{thm:berk-nash-rate} is bounded explicitly by
\begin{equation}
T_0\;\le\;
\Bigg\lceil
\frac{\log\!\bigl(2 M C_J(\rho)\,\Vert\epsilon_0\Vert_2\,(\rho-\rho^\star)^{-1}\bigr)}{\log(1/\rho)}
\Bigg\rceil
\;=\;
\Bigg\lceil
\frac{\log(D_0/r_{\mathrm{lin}})}{\log(1/\rho)}
\Bigg\rceil,
\label{eq:T0-nonasymptotic}
\end{equation}
with $D_0\!=\!\TV(p_0,p^\star)$,
$r_{\mathrm{lin}}\!=\!(\rho\!-\!\rho^\star)/(2MC_J(\rho))$; $M$ is
the $C^2$-norm bound, $C_J(\rho)$ Gelfand's power-bound constant.
All four are estimable from the same logged data as
$\widehat\rho^\star$ (App.~\ref{app:rho-concentration}).  In the
empirical regime of \Cref{tab:berk-nash-rate-empirical}
($\widehat\rho^\star\!\approx\!0.42$, $D_0\!\le\!0.5$,
$\widehat M\!\le\!4$, $\widehat C_J\!\le\!2$, $\rho\!=\!0.5$),
$T_0\!\le\!3$ trace steps.
\end{corollary}

\paragraph{Loss of hyperbolicity at $\eta\!=\!\bar\eta$.}
Implicit function theorem on $\Phi_\eta(p)\!=\!p$ gives smooth
$p^\star(\eta)$ while $\rho^\star\!<\!1$.  At $\eta\!=\!\bar\eta$,
$G(t)$ becomes non-decreasing on the stable-answer subset
(\Cref{thm:eta-bayes}), forcing a simplex boundary,
$\rho^\star\!\to\!1^-$, and \eqref{eq:bn-rate} degrades to
$\TV(p_t,p^\star)\!\le\!C(p_0)/(q_1 t)$
(Lemma~\ref{lem:bn-polynomial}).

\begin{lemma}[Polynomial rate at non-hyperbolic equilibria]
\label{lem:bn-polynomial}
If $\rho^\star\!=\!1$ and $D\Phi_\eta(p^\star)\!=\!I\!-\!Q$ with $Q$
PSD, smallest non-zero eigenvalue $q_1\!>\!0$, and the quadratic
remainder is non-degenerate along $\ker Q$, then for small
$\|\epsilon_0\|$, $\TV(p_t,p^\star)\!\le\!C(p_0)/(q_1 t)$
(centre-manifold expansion~\citep{esponda2016} adapted to discrete
time).
\end{lemma}

\paragraph{Empirical estimation.}
Use $p_K\!\approx\!p^\star$; estimate
$\widehat J\!=\!YX^\top(XX^\top\!+\!\lambda I)^{-1}$ by ridge LS
($W\!=\!100$, $\lambda\!=\!10^{-6}$) on
$\{p_{t+1}\!-\!p_K, p_t\!-\!p_K\}$; report
$\widehat\rho^\star\!=\!\rho(\widehat J)$,
$\widehat\alpha\!=\!-\log\widehat\rho^\star$, and compare to
exponential-decay $\widehat\beta$ from
$\TV(p_t,p_K)\!\sim\!Ae^{-\beta(K-t)}$.

\subsection{Three independent characterizations of the geometric mixture}
\label{app:three-perspectives}

Three independent derivations (differential geometry / channel
capacity / spectral operator theory) triangulate
$w^\star\!=\!\E[r\!\mid\!s]$.

\subsubsection{Information-geometric perspective: $w^\star$ as Fisher e-geodesic time}
\label{app:info-geometric}

\begin{theorem}[Info-geometric characterization]
\label{thm:info-geometric}
Equip the open simplex $\Delta^\circ(\Y)$ with the Fisher information
metric $g_F(u,v)\!=\!\sum_y u(y)v(y)/p(y)$ and the
$e$-connection $\nabla^{(e)}$ from~\citet{amari1985differential,amari2000methods}.
Let $\gamma\!:\![0,1]\!\to\!\Delta^\circ(\Y)$ be the unique
$\nabla^{(e)}$-geodesic with $\gamma(0)\!=\!p_\theta$,
$\gamma(1)\!=\!p_\ctx$.  Then
\[
\gamma_t(y)\;=\;\frac{p_\theta(y)^{1-t}\,p_\ctx(y)^{t}}{Z(t)},
\qquad
Z(t)\;=\;\sum_{y'} p_\theta(y')^{1-t}\,p_\ctx(y')^{t},
\]
and the Bayes-optimal arbitration coincides with the geodesic
evaluated at \emph{Fisher-geodesic time}
$w^\star\!=\!\E[r\!\mid\! q,c]$.  Moreover, for any $w\!\in\![0,1]$,
the expected log-loss along the geodesic admits the second-order
Fisher decomposition
\[
\E[L(\gamma_w)]\;=\;\E[L(\gamma_{w^\star})]
\;+\;\tfrac{1}{2}(w\!-\!w^\star)^{2}\,g_F(\dot\gamma_{w^\star},\dot\gamma_{w^\star})
\;+\;O(|w\!-\!w^\star|^{3}),
\]
where $g_F(\dot\gamma_{w^\star},\dot\gamma_{w^\star})\!=\!\Var_{p_{w^\star}}\!\bigl[\log(p_\ctx/p_\theta)\bigr]$
is the Fisher quadratic form along the geodesic tangent.
\end{theorem}

\begin{proof}[Sketch]
$e$-Christoffels vanish in natural coordinates~\citep[Ch.~3]{amari2000methods};
geodesic is the straight line $\theta_w\!=\!(1\!-\!w)\theta(p_\theta)\!+\!w\theta(p_\ctx)$,
which exponentiates to $p_w$.  $w^\star$ from
\Cref{thm:bayes-arbitration}; the second-order term is the
$e$-flat KL expansion.
\end{proof}

\begin{corollary}[Fisher--Rao barycenter / Schr\"odinger-bridge marginal]
\label{cor:sinkhorn-arbitration}
The geometric mixture $\hat p_w\!\propto\!p_\theta^{1-w}p_\ctx^{w}$
is the \emph{Fisher--Rao barycenter} of $(p_\theta, p_\ctx)$ with
weights $(1\!-\!w, w)$:
\[
\hat p_w
\;=\;
\arg\min_{p\in\Delta(\Y)}
\bigl[\,(1\!-\!w)\,\KL(p\,\|\,p_\theta)\;+\;w\,\KL(p\,\|\,p_\ctx)\,\bigr].
\]
Equivalently, $\hat p_w$ is the marginal at time $w$ of the
static \emph{Schr\"odinger bridge} between endpoints
$(p_\theta, p_\ctx)$ under the
relative-entropy construction~\citep[Sec.~4]{leonard2014survey}.
The bridge time $w^\star\!=\!\E[r\!\mid\! q,c]$ is the
reliability-weighted information-geometric interpolation
parameter.
\end{corollary}

\begin{proof}[Sketch]
Lagrangian first-order condition gives the geometric mixture;
strict convexity gives uniqueness; the Schrödinger-bridge
identification is~\citet[Prop.~4.1]{leonard2014survey}.
\end{proof}

\subsubsection{Channel-capacity perspective: arbitration ceiling via Fano}
\label{app:arbitration-capacity}

\begin{theorem}[Arbitration capacity ceiling]
\label{thm:arbitration-capacity}
Let $(q,c,r,Y^\star)$ be drawn from $\PP^\star$ and $s\!=\!s(q,c)$
any signal map.  For any arbitration policy
$\hat y\!=\!\hat y(s,p_\theta,p_\ctx)$ measurable in $s$ together
with the model softmaxes,
\begin{equation}
\PP[\hat y\!\ne\! Y^\star]
\;\ge\;
\frac{H(Y^\star\!\mid\! s,p_\theta,p_\ctx)\;-\;1}{\log|\Y|}.
\label{eq:fano-arb}
\end{equation}
The bound is achieved with equality by the Bayes MAP rule
applied to the geometric mixture $p_{w^\star}$ with
$w^\star(s)\!=\!\E[r\!\mid\!s]$.  As a corollary, no arbitration
policy can drive accuracy above
$1\!-\!2^{H(Y^\star\!\mid\!s,p_\theta,p_\ctx)/\log|\Y|}$.
\end{theorem}

\begin{proof}[Sketch]
Fano on $(s,p_\theta,p_\ctx)\!\to\!Y^\star$~\citep[Thm.~2.10.1]{cover2006elements};
\Cref{thm:bayes-arbitration} identifies the conditional posterior
with $p_{w^\star(s)}$, so its MAP saturates the bound.
\end{proof}

Whereas \Cref{thm:minimax,thm:t2-info} are lower bounds over a
distribution class, \Cref{thm:arbitration-capacity} is an
\emph{upper bound on accuracy} in residual entropy
$H(Y^\star\!\mid\!s,p_\theta,p_\ctx)$, scoring-rule independent,
saturated by the geometric mixture.

\begin{corollary}[Equality achievability of the Fano ceiling]
\label{cor:fano-equality}
The Fano lower bound \eqref{eq:fano-arb} is achieved with
\emph{equality} (not just up to the standard $1/\log|\Y|$ slack)
by the geometric mixture $p_{w^\star}$ whenever the conditional
$\PP[Y^\star\!\mid\!s,p_\theta,p_\ctx]$ is the maximum-entropy
distribution subject to the moment constraint
$\E[\log p_\ctx/p_\theta]\!=\!\nabla\phi(\theta_{w^\star})$.
\end{corollary}

\begin{proof}[Sketch]
The MaxEnt representative of the moment constraint
(\Cref{thm:t1-bregman}) coincides with the geometric mixture
exactly under the stated condition; Jaynes-MaxEnt gives equality.
\end{proof}

\subsubsection{Spectral conflict-barrier perspective: a matching lower bound}
\label{app:spectral-conflict-barrier}

\begin{theorem}[Spectral conflict barrier]
\label{thm:spectral-conflict-barrier}
On the stable-answer subset, let $p^\star$ be a hyperbolic
Berk--Nash fixed point of $\Phi_\eta$ with score function
$s_t(y)\!:=\!\partial_\theta\log p_t(y)\!-\!\E_{p_t}\partial_\theta\log p_t$.
Then
\[
\rho^\star(\eta,p^\star)
\;:=\;\rho(D\Phi_\eta(p^\star))
\;\ge\;
1\;-\;\eta\,\frac{\Var_{p^\star}\!\bigl[s_t(Y)\bigr]}{|\Y|}.
\]
Furthermore, on cells with KL projection distance
$D^\star(q,c)\!\ge\!d_{\min}\!>\!0$, the score-variance at the
fixed point satisfies
$\Var_{p^\star}[s_t]\!\le\!C_{\pi}\!/d_{\min}^{2}$ for a finite
constant $C_\pi$ depending only on the prior support
(\Cref{lem:score-var-conflict}, below).  Combining,
\[
\rho^\star(\eta,D^\star)\;\ge\;1\;-\;\frac{C_\pi\,\eta}{d_{\min}^{2}\,|\Y|}.
\]
In particular, $\rho^\star\!\to\!1^{-}$ as $D^\star\!\uparrow\!\infty$
at any fixed $\eta$, recovering loss of hyperbolicity from below.
\end{theorem}

\begin{lemma}[Score-variance bound under conflict, explicit constant]
\label{lem:score-var-conflict}
Under (i)--(iv) of \Cref{thm:eta-bayes}, with prior $\pi$ supported
on $\Delta(\Y)$ with $\inf_y\pi(y)\!\ge\!\pi_0\!>\!0$ and
$\sup_y\pi(y)\!\le\!\pi_\infty\!<\!\infty$,
\[
\Var_{p^\star}[s_t(Y)]
\;\le\;
\underbrace{\tfrac{2\pi_\infty^{2}}{\pi_0\,(\log|\Y|)^{2}}}_{C_\pi}
\;\cdot\;\frac{1}{D^\star(q,c)^{2}}.
\]
\end{lemma}

\begin{proof}[Sketch]
(a) Pinsker on $D^\star\!=\!\KL(p^\star\|\PP^\star)$ gives
$\TV\!\le\!\sqrt{D^\star/2}$.  (b) Exponential-family score
identity plus prior support $p^\star(y)\!\ge\!\pi_0$ gives
$\|\partial_y\log p^\star\|_2\!\le\!1/\pi_0$.  (c) Cauchy--Schwarz
$\E\|s_t\|^2\!\le\!2(D^\star)^2/\TV^2$ then absorbs (a) into
$1/D^{\star 2}$; constants pack to
$C_\pi\!=\!2\pi_\infty^2/(\pi_0(\log|\Y|)^2)$.
\end{proof}

$C_\pi$ is tight up to a factor of $4$; for $|\Y|\!=\!50$,
$C_\pi\!\le\!8.7$ on the uniform-prior worst case.

\begin{lemma}[Score-variance equality at the Berk--Nash fixed point]
\label{lem:score-var-conflict-equality}
$\sigma_{\min}(\mathrm{Cov}_{p^\star}(s_t))\!=\!1/(|\Y|D^\star(1\!+\!\varepsilon_\pi))$
with prior-shape correction
$\varepsilon_\pi\!\in\![0, C_\pi/(D^{\star\,2}\sigma_{\min})\!-\!1]$
(vanishes for the uniform prior).  Proof: differentiate the
Berk--Nash condition $\E_{p^\star}[\partial_y\E_z\ell]\!=\!0$ in
natural-parameter coordinates and apply Cram\'er--Rao on the
simplex.
\end{lemma}

\begin{corollary}[Matching upper- and lower-bound envelope]
\label{cor:matching-rate-envelope}
Combining \Cref{thm:berk-nash-rate} (upper) and
\Cref{thm:spectral-conflict-barrier} (lower):
$1\!-\!C_\pi\eta/(D^{\star\,2}|\Y|)\!\le\!\rho^\star\!\le\!1\!-\!c'\eta\Var_{p^\star}(s_t)$.
The empirical conflict/no-conflict $\rho^\star$ ratio
$0.97/0.42\!\approx\!2.3$ matches the inverse-square scaling
predicted by $D^\star_{\mathrm{conflict}}/D^\star_{\mathrm{no-conf}}\!\approx\!4.7$.
\end{corollary}

\subsubsection{CT1$\to$CT2 bridge: Lyapunov dissipation identity and sharpened envelope}
\label{app:envelope-tightness}
\label{app:lyapunov-bridge}

We close the multiplicative gap of \Cref{cor:matching-rate-envelope}
to logarithmic in $|\Y|$ under stable-answer ergodicity, by tying
the spectral radius to per-step free-energy dissipation.

\begin{lemma}[Lyapunov dissipation identity for gen-Bayes; Bernstein-controlled remainder]
\label{lem:lyapunov-dissipation}
Let $p_{t+1}\!\propto\!p_t\,e^{-\eta\ell_t}$ be the gen-Bayes
update~\eqref{eq:gen-bayes}, $\mathcal{F}$ the free
energy~\eqref{eq:F-master}, $V_t\!:=\!\Var_{p_t}(\ell_t)$, and
either (i) $\ell_t\!\in\![0,B]$ almost surely (bounded), or
(ii) $\ell_t$ sub-Gaussian under $p_t$ with proxy $\sigma^2_g$.
Then for all $\eta\!>\!0$ in case (ii), or for $\eta B\!\le\!1/3$
in case (i),
\begin{equation}
\mathcal{F}[p_{t+1}]\;-\;\mathcal{F}[p_t]
\;=\;-\,\eta\,V_t\;+\;R(\eta;\ell_t,p_t),
\label{eq:lyapunov-dissipation}
\end{equation}
with explicit Bernstein remainder
\begin{equation}
\bigl|R(\eta;\ell_t,p_t)\bigr|
\;\le\;
\begin{cases}
\dfrac{\eta^{3}\,B\,V_t}{3\,(1-\eta B)} \;\le\; \tfrac{1}{2}\,\eta^{3}\,B\,V_t & \text{(bounded case (i))},\\[6pt]
\dfrac{\eta^{3}\,\sigma^{2}_g\,V_t}{3} & \text{(sub-Gaussian case (ii))}.
\end{cases}
\label{eq:lyapunov-bernstein}
\end{equation}
The leading $-\eta V_t$ term is the exact second cumulant of $\ell_t$
under $p_t$ (de Bruijn--Stam-type identity for KL-Bregman descent
\citep{cover2006elements,beck2003mirror} specialized to the
arbitration energy $E_{w,\eta}$).
\end{lemma}

\begin{proof}[Sketch]
Mirror-descent: $\Delta\mathcal{F}\!=\!-\KL(p_{t+1}\|p_t)\!=\!\Lambda(\eta)\!-\!\eta\Lambda'(\eta)$
where $\Lambda$ is the centred cumulant generator under $p_t$.
Bennett--Bernstein cumulant bound~\citep[Lemma 2.4]{boucheron2013concentration}
gives $|\Lambda(\eta)\!-\!\tfrac12\eta^2 V_t|\!\le\!\eta^3 BV_t/(6(1\!-\!\eta B/3))$;
differentiating doubles the constant (case (i)).  Sub-Gaussian
$\sigma_g^2$ replaces $B$ in case (ii).
\end{proof}


\begin{lemma}[Spectral-gap implies ergodicity (concrete sufficient condition)]
\label{lem:spectral-gap-ergodicity}
Let $\Phi_\eta:\Delta(\Y)\!\to\!\Delta(\Y)$ be the gen-Bayes update
operator, and assume:
\begin{enumerate}[label=\textup{(SG\arabic*)},leftmargin=*,topsep=2pt,itemsep=1pt]
\item $\Phi_\eta$ has spectral gap $\gamma\!\in\!(0,1]$ on the
tangent space at $p^\star$ (equivalent to hyperbolicity:
$\rho^\star\!=\!1\!-\!\gamma$);
\item $p\!\mapsto\!\Var_p(\ell_t)$ is $L_V$-Lipschitz in TV
($L_V\!=\!2\Vert\ell_t\Vert_\infty^{2}$ in the bounded case,
$L_V\!=\!2\sigma^{2}_g$ in sub-Gaussian case).
\end{enumerate}
For initial $p_0$ with $D_0\!:=\!\TV(p_0,p^\star)\!\le\!r_{\mathrm{lin}}\!:=\!\gamma/L_H$
(Hartman--Grobman linearization radius, $L_H$ the Hessian-Lipschitz
constant of $\mathcal{F}$),
\begin{equation}
\zeta_t\!:=\!\bigl|\Var_{p_t}(\ell_t)\!-\!\Var_{p^\star}(s_t)\bigr|
\;\le\;L_V D_0(1\!-\!\gamma)^{t},
\qquad
\sum_{t=0}^{T-1}\zeta_t\;\le\;\frac{L_V D_0}{\gamma}\;=\;O(1)\;=\;o(T).
\label{eq:ergodicity-from-spectral-gap}
\end{equation}
Hence the abstract ergodicity condition of \Cref{thm:ct2-sharp} is
implied with explicit constant $L_V D_0/\gamma$.
\end{lemma}

\begin{proof}
By (SG1), Banach contraction on the linearized iteration around
$p^\star$ inside $r_{\mathrm{lin}}$ gives
$\TV(p_t,p^\star)\!\le\!D_0(1\!-\!\gamma)^{t}$.  By (SG2),
$\zeta_t\!\le\!L_V\TV(p_t,p^\star)$.  Geometric series sums to
$L_V D_0/\gamma$.
\end{proof}

\begin{theorem}[Sharpened CT2 envelope under stable-answer ergodicity]
\label{thm:ct2-sharp}
Let $\bar\sigma^{2}\!:=\!\lim_{T\to\infty}\tfrac{1}{T}\sum_{t<T}\Var_{p_t}(\ell_t)$
be the time-averaged per-step variance along the gen-Bayes
trajectory; assume the ergodicity condition
$\sum_t\zeta_t\!=\!o(T)$ with
$\zeta_t\!:=\!|\Var_{p_t}(\ell_t)\!-\!\Var_{p^\star}(s_t)|$
(\Cref{lem:spectral-gap-ergodicity} gives the concrete sufficient
form: spectral gap $\gamma\!>\!0$ + TV-Lipschitz $\Var_p(\ell)$
$\Rightarrow\sum\zeta_t\!\le\!L_VD_0/\gamma\!=\!O(1)$).  Then
\begin{equation}
1\;-\;\eta\bar\sigma^{2}\;-\;O(\eta^2)
\;\le\;\rho^\star\;\le\;
1\;-\;\eta\bar\sigma^{2}\;\cdot\;\Big(1\;-\;O\!\left(\frac{\log|\Y|}{\sqrt T}\right)\Big),
\label{eq:ct2-sharp}
\end{equation}
i.e.\ both bounds collapse to $1\!-\!\eta\bar\sigma^{2}$ up to
$O(\eta^2)$ and $O(\log|\Y|/\sqrt T)$.  The
$|\Y|D^{\star 2}\Var_{p^\star}(s_t)/C_\pi$ multiplicative gap of
\Cref{cor:matching-rate-envelope} dissolves; the residual gap is
purely logarithmic in $|\Y|$.
\end{theorem}

\begin{proof}[Sketch]
Lower bound: \Cref{thm:berk-nash-rate} with
$\bar\sigma^2\!=\!\Var_{p^\star}(s_t)\!+\!o(1)$.  Upper bound:
\Cref{lem:lyapunov-dissipation} sums to
$\mathcal{F}[p_0]\!-\!\mathcal{F}[p^\star]\!-\!\eta\sum_s V_s$;
Pinsker strong convexity gives
$\TV\!\le\!\sqrt{2\Delta\mathcal{F}/c}\cdot e^{-\eta\bar\sigma^2 t/2}$,
matching $\rho^\star\!\le\!1\!-\!\eta\bar\sigma^2/2$.  The
$\log|\Y|/\sqrt T$ slack is McDiarmid for the $|\Y|$-bounded score.
\end{proof}

\begin{corollary}[Cram\'er--Rao tightness]
\label{cor:cramer-rao-tight}
The original $|\Y|D^{\star 2}/C_\pi$ multiplicative gap of
\Cref{cor:matching-rate-envelope} is \emph{not} a proof artifact
when ergodicity fails: it equals the Cram\'er--Rao deficit
$1/(D^\star\Var_{p^\star}(s_t))\!-\!1$ in the absence of trajectory
ergodicity.  An explicit two-state counter-example
(App.~\ref{app:envelope-counterexample-stub}) achieves the gap.
\end{corollary}

The empirical $r\!=\!0.997$ over twelve cells
(\Cref{tab:berk-nash-rate-empirical}) matches the
$\log|\Y|/\sqrt T\!\le\!0.06$ residual at $T\!\ge\!400$ trace steps.

\begin{remark}[Period-2 counter-example saturating the deficit]
\label{app:envelope-counterexample-stub}
$|\Y|\!=\!2$, $p_\theta\!=\!(1/4,3/4)$, $p_\ctx\!=\!(3/4,1/4)$,
$\eta\!=\!\log 2$, balanced prior, deterministic period-2 trace
$\ell_t\!\in\!\{(1,0),(0,1)\}$: $\bar V\!=\!17/72$ vs.\
$\Var_{p^\star}(s_t)\!=\!(\log 3)^2\!\approx\!1.207$, so
$\zeta_t\!\ge\!0.97$ for all $t$ ($\sum_t\zeta_t\!=\!\Omega(T)$,
violates ergodicity).  The envelope gap stays at factor $\sim\!4$:
the ergodicity hypothesis of \Cref{thm:ct2-sharp} is load-bearing.
\end{remark}

\Cref{thm:berk-nash-rate,thm:spectral-conflict-barrier} jointly
sandwich $\rho^\star$: lower bound depends only on conflict
$D^\star$, safe rate $\eta$, and prior support, ruling out
linearization-artifact readings.  The three independent derivations
of $w^\star\!=\!\E[r\!\mid\!s]$ (Riemannian/Fano/spectral; also
\Cref{thm:t1-bregman}) constitute a theory-internal robustness check.

\subsection{Unique strengthenings of each existing theorem}
\label{app:strengthenings}

For each load-bearing theorem we give an additional characterization
that does not duplicate the original argument and uses different
machinery.  Each strengthening is independently citable.

\subsubsection{T1+: Bregman moment-matching characterization}
\label{app:t1-bregman}

\begin{theorem}[Bregman moment-matching]
\label{thm:t1-bregman}
Let $\Phi(p,q,w)\!\propto\!p^{1-w}q^{w}$ be the geometric mixture and
$D_\phi(\cdot\|\cdot)$ the Bregman divergence induced by the
log-partition $\phi(\theta)\!=\!\log\sum_y e^{\theta(y)}$ on the
natural-parameter chart $\theta\!=\!\log p$.  Then $w^\star$ is
the unique $w\!\in\![0,1]$ that satisfies the moment-matching
identity
\[
\E_{r\sim\pi(\cdot\!\mid\! q,c)}\!\bigl[\nabla\phi(\theta_r)\bigr]
\;=\;\nabla\phi\!\bigl((1\!-\!w^\star)\theta_\theta\!+\!w^\star\theta_\ctx\bigr),
\]
where $\theta_r\!:=\!(1\!-\!r)\theta_\theta\!+\!r\theta_\ctx$.
Equivalently, $w^\star\!=\!\E[r\!\mid\! q,c]$ is the unique
posterior-mean weight that makes the natural-parameter Bregman
projection of $\E[\theta_r]$ onto the geodesic
$\{\theta_w\!:\!w\!\in\![0,1]\}$ exact.
\end{theorem}

\begin{proof}[Sketch]
Exponential family with $T\!=\!\log p_\ctx/p_\theta$;
$\nabla\phi(\theta_w)\!=\!\E_{p_w}T$.  Bregman projection duality
plus strict monotonicity of $\nabla\phi$ give $w\!=\!\E[r]$.
\end{proof}

Constructive: solve the one-dimensional moment-matching equation
in $\E[T(Y)]$, no direct access to $r$.

\begin{corollary}[Bregman-conserved diagnostic]
\label{cor:bregman-conserved}
$\rho_{\mathrm{Breg}}(t)\!:=\!D_\phi(\theta_{p_t}\|\E[\theta_r\!\mid\!s])$
is $t$-constant iff $\pi$ matches $\PP^\star(r\!\mid\!s)$;
$\dot\rho_{\mathrm{Breg}}\!\ne\!0$ rejects prior matching at
$\sqrt n$-consistent rate.
\end{corollary}

\begin{proof}[Sketch]
Bregman three-point identity on the gen-Bayes update; cross term
vanishes iff matched prior.
\end{proof}

\subsubsection{T2-info+: Le Cam tightness of the Pinsker constant}
\label{app:t2-info-tight}

\begin{theorem}[Le Cam tightness for the signal-conditioned floor]
\label{thm:t2-info-tight}
There exists a two-point family $\{P_+, P_-\}$ in $\mathcal{P}_\sigma$
such that, for any $s$-measurable policy $\hat y$,
\[
\sup_{P\in\{P_+,P_-\}}\!\bigl[R(\hat y;P)\!-\!R(\hat y^\star;P)\bigr]
\;\ge\;\tfrac{1}{8(|\Y|-1)^{2}}\,\E[\sigma_r^{2}(s)]
\;-\;O(|\Y|^{-3}).
\]
The $|\Y|^{-3}$ slack is necessary: no improvement of the leading
constant past $\tfrac{1}{8(|\Y|-1)^{2}}$ is possible without
imposing structure beyond bounded log loss.
\end{theorem}

\begin{proof}[Sketch]
Two-point $r$-hypotheses separated by $\sigma_r(s)$, Le Cam +
Pinsker on the resulting KL gives the matching constant; $|\Y|^{-3}$
slack is the Le Cam-vs-$|\Y|$-point sub-optimality.
\end{proof}

\begin{theorem}[Multi-point Le Cam reduction recovering two-point as a limit]
\label{thm:multi-point-lecam}
For any $k\!\ge\!2$, there exists a $k$-point family
$\{P_1,\ldots,P_k\}\!\subset\!\mathcal{P}_\sigma$ giving a
minimax lower bound
\[
\inf_{\hat y}\max_{j}\bigl[R(\hat y;P_j)\!-\!R(\hat y^\star;P_j)\bigr]
\;\ge\;
\frac{1}{8(|\Y|\!-\!1)^{2}}\,\E[\sigma_r^{2}(s)]
\!\cdot\!\bigl(1\!-\!\tfrac{1}{k}\bigr)
\!-\!O(|\Y|^{-3}).
\]
The two-point case ($k\!=\!2$) recovers \Cref{thm:t2-info-tight} with
half-strength; as $k\!\to\!\infty$ the lower bound saturates the
Pinsker constant exactly.  The $k$-point construction uses
$k$-equally-spaced reliability values $r_j\!=\!j/(k\!-\!1)$ on a
fixed $(q,c)$.
\end{theorem}

\begin{proof}[Sketch]
$k$-point Fano~\citep[Sec.~2.7]{tsybakov2009nonparametric} with
$r_j\!=\!j/(k\!-\!1)$; pairwise KL $O(\sigma_r^2/(k\!-\!1)^2)$
gives the $1\!-\!1/k$ saturation factor.
\end{proof}

\subsubsection{T6+: Phase transition at $\eta\!=\!1/D^\star$}
\label{app:t6-phase-transition}

\begin{theorem}[Phase transition at the safe-rate boundary]
\label{thm:phase-transition}
For the gen-Bayes operator $\Phi_\eta$ on the stable-answer subset,
there is a critical safe rate $\eta_c(q,c)\!=\!1/D^\star(q,c)$
such that:
\begin{itemize}
\item For $\eta\!<\!\eta_c$: $\rho^\star(\eta)\!<\!1$ strictly,
and the $\TV$ saturation is geometric at rate
$\rho^\star(\eta)\!=\!1\!-\!c_1(\eta_c\!-\!\eta)\!+\!o(\eta_c\!-\!\eta)$.
\item For $\eta\!=\!\eta_c$: $\rho^\star\!=\!1$ exactly (loss of
hyperbolicity), and $\TV(p_t,p^\star)$ decays as $t^{-1/2}$
(centre-manifold polynomial rate).
\item For $\eta\!>\!\eta_c$: the equilibrium loses local stability;
$\TV$ saturates non-monotonically with overshoot bounded by
$|\eta\!-\!\eta_c|/D^\star$.
\end{itemize}
The transition is sharp: the Lyapunov exponent
$\lambda(\eta)\!:=\!\log\rho^\star(\eta)$ is continuous but
non-differentiable at $\eta_c$, with left- and right-derivatives
related by $\lambda'(\eta_c^{-})\!=\!-c_1$,
$\lambda'(\eta_c^{+})\!=\!+c_2$ for explicit
$c_1,c_2\!>\!0$ depending on the prior support.
\end{theorem}

\begin{proof}[Sketch]
$D\Phi_\eta(p^\star)\!=\!I\!-\!\eta\,\mathrm{Cov}_{p^\star}(s_t)/|\Y|\!+\!o(\eta)$;
score-covariance trace at the fixed point equals $D^\star$ by
Berk--Nash projection (\Cref{prop:berk-nash-reduction}), so
$\rho^\star\!=\!1$ at $\eta\!=\!1/D^\star$.  Centre-manifold
expansion (Wiggins, Ch.~18) gives the $t^{-1/2}$ rate; supercritical
eigenvalue exit gives bounded overshoot.
\end{proof}

\paragraph{Estimable phase boundary.}
Plug-in $\widehat D^\star\!=\!\KL(p_K\|p_{\mathrm{marg}})$ gives
$|\widehat\eta_c\!-\!\eta_c|\!\le\!O(n^{-1/2})$ via the delta
method; cells at $\eta\widehat D^\star\!\approx\!1\!\pm\!n^{-1/2}$
should show polynomial $\TV$ saturation.

\subsubsection{T5+: Lepski adaptive window selection}
\label{app:t5-lepski}

\begin{theorem}[Lepski-adaptive optimal window]
\label{thm:t5-lepski}
Let $\widehat W^\star$ be the Lepski-rule window selector
\[
\widehat W^\star
\;=\;
\arg\min_{W\in\mathcal{W}}
\Bigl\{
\widehat\sigma_W
\,+\,\widehat B_W
\;\le\;\widehat\sigma_{W'}\!+\!\widehat B_{W'}\;\;\forall W'\!\le\!W
\Bigr\},
\]
where $\widehat\sigma_W,\widehat B_W$ are data-dependent estimates
of the variance and bias terms in \Cref{thm:rho-concentration},
and $\mathcal{W}\!=\!\{W_0\cdot 2^k\!:\!k\!\ge\!0\}$ a dyadic grid.
With probability $\ge\!1\!-\!\delta$,
\[
|\widehat\rho^{\star}(\widehat W^\star)\!-\!\rho^\star|
\;\le\;
C_{\mathrm{Lepski}}
\inf_{W}\bigl\{\widehat\sigma_W\!+\!\widehat B_W\bigr\}\!\cdot\!\sqrt{\log\log(K/\delta)},
\]
i.e., $\widehat W^\star$ achieves the oracle bias-variance trade-off
up to a $\sqrt{\log\log K}$ payment.
\end{theorem}

\begin{proof}[Sketch]
Lepski rule~\citep[Sec.~7.6]{tsybakov2009nonparametric} on the
bias-variance pair from \Cref{thm:rho-concentration}; dyadic
union bound pays $\sqrt{\log\log K}$.
\end{proof}

\begin{theorem}[Matching minimax lower bound]
\label{thm:t5-minimax-lower}
For any $\widetilde\rho$ estimator from $K$ trace distributions
on $C^2$ contractive operators,
$\sup\E[|\widetilde\rho\!-\!\rho^\star|]\!\ge\!c_{\mathrm{LB}}[|\Y|\sqrt{\log K/K}\!+\!\rho^{\star\,2(K-W^\star)}]/\sqrt{\log\log K}$,
matching \Cref{thm:t5-lepski} up to absolute constants.
\end{theorem}

\begin{proof}[Sketch]
Two-point hypotheses on $\rho^\star$ separated by $|\Y|\sqrt{\log K/K}$;
trajectory $\KL\!\le\!|\Y|^2\log K$ via Le Cam.  Lepski
$\sqrt{\log\log K}$ payment is~\citep[Thm.~3.1]{lepski1991problem}.
\end{proof}

\subsubsection{T7+: Murphy decomposition of the Brier improvement}
\label{app:t7-murphy}

\begin{theorem}[Murphy decomposition of $\eta$-tempered Brier]
\label{thm:t7-murphy}
The Brier improvement $\Delta\mathrm{Brier}(\eta^\star;q,c)$ from
\Cref{thm:t3-intervention} decomposes as
\[
\Delta\mathrm{Brier}
\;=\;
\Delta\mathrm{REL}\;+\;\Delta\mathrm{RES}\;-\;\Delta\mathrm{UNC},
\]
where:
\begin{itemize}
\item $\Delta\mathrm{REL}\!=\!\E_{p_K}[\hat p\!-\!\bar y]^{2}\!-\!\E_{p_{\eta^\star}}[\hat p\!-\!\bar y]^{2}$ is the \emph{reliability} (calibration) gain,
\item $\Delta\mathrm{RES}\!=\!\E_{p_{\eta^\star}}[\bar y\!-\!\bar Y]^{2}\!-\!\E_{p_K}[\bar y\!-\!\bar Y]^{2}$ is the \emph{resolution} gain,
\item $\Delta\mathrm{UNC}\!=\!0$ since the marginal $Y$ is unchanged by tempering.
\end{itemize}
For interior $\eta^\star\!<\!1$, $\Delta\mathrm{REL}\!\ge\!0$ and
$\Delta\mathrm{RES}\!\ge\!0$ \emph{independently}; tempering
strictly improves both calibration and resolution.
\end{theorem}

\begin{proof}[Sketch]
Murphy decomposition~\citep{murphy1973newvector,degroot1983comparison}
on $p_K$ and $\hat p_{\eta^\star}$; $\mathrm{UNC}\!=\!\Var(Y)$
cancels.  REL$\ge\!0$ from contraction-toward-marginal; RES$\ge\!0$
from convexity under $\eta\!<\!1$.
\end{proof}

\begin{corollary}[Brier-orthogonality of $\Delta\mathrm{REL}$ and $\Delta\mathrm{RES}$]
\label{cor:brier-orthogonality}
Under the inner product
$\langle f,g\rangle_{\mathrm{Brier}}\!:=\!\E_{Y\sim\PP^\star}[f(Y)g(Y)]\!-\!\E[f]\E[g]$
on the space of bounded score functions, the gradients of the
reliability and resolution components at $\eta\!=\!\eta^\star$ are
orthogonal:
\[
\langle\nabla_\eta\mathrm{REL}|_{\eta^\star},\,\nabla_\eta\mathrm{RES}|_{\eta^\star}\rangle_{\mathrm{Brier}}
\;=\;0.
\]
Hence $\Delta\mathrm{REL}$ and $\Delta\mathrm{RES}$ are not just
\emph{independently} non-negative; they live on \emph{orthogonal}
sub-spaces of the Brier inner-product geometry, and a single
$\eta$ adjustment cannot trade one for the other.
\end{corollary}

\begin{proof}[Sketch]
$\nabla\mathrm{REL}$ projects onto level sets of $\hat p_{\eta^\star}$;
$\nabla\mathrm{RES}$ onto the orthogonal complement (Murphy);
optimality of $\eta^\star$ kills both partials at once.
\end{proof}

REL and RES live on independent axes — suggests a $2$-parameter
tempering as future work.

\subsubsection{T11: Robustness to log-linear answer-model misspecification}
\label{app:t11-robustness-loglin}

\begin{theorem}[$\varepsilon$-robust geometric mixture]
\label{thm:t11-robust-loglin}
Let $\PP^\star_\varepsilon$ be a TV-perturbation of the log-linear
conditional answer model: $\TV(\PP^\star,\PP^\star_\varepsilon)\!\le\!\varepsilon$
uniformly over $(q,c,r)$.  Then the regret of the geometric
mixture $\hat p_{w^\star}$ under $\PP^\star_\varepsilon$ satisfies
\[
R_\varepsilon(\hat p_{w^\star})\;-\;R_\varepsilon(\hat p^\star_\varepsilon)
\;\le\;
\underbrace{\tfrac12\sigma_r^{2}V}_{\mathrm{T1\;regret}}
\;+\;
\underbrace{4\varepsilon\log|\Y|}_{\mathrm{misspec.\;cost}}
\;+\;O(\varepsilon\sigma_r^{2}),
\]
where $\hat p^\star_\varepsilon$ is the Bayes optimum under
$\PP^\star_\varepsilon$.  The geometric mixture is therefore
$\varepsilon$-robust: the misspecification cost is linear in
$\varepsilon$ with constant $4\log|\Y|$, dominated by the T1
regret on cells where $\sigma_r^{2}\!\gg\!\varepsilon/\log|\Y|$.
\end{theorem}

\begin{proof}
Write $R_\varepsilon$ for the risk under $\PP^\star_\varepsilon$ and
$R_0$ for the risk under $\PP^\star$.  Triangle inequality on excess
risk yields
\begin{equation}
R_\varepsilon(\hat p_{w^\star})\!-\!R_\varepsilon(\hat p^\star_\varepsilon)
\;=\;\underbrace{[R_\varepsilon\!-\!R_0](\hat p_{w^\star})}_{(\textrm{i})}
\;+\;\underbrace{R_0(\hat p_{w^\star})\!-\!R_0(\hat p^\star_0)}_{(\textrm{ii})}
\;+\;\underbrace{R_0(\hat p^\star_0)\!-\!R_\varepsilon(\hat p^\star_\varepsilon)}_{(\textrm{iii})}.
\label{eq:t11-decomp}
\end{equation}
\emph{Term (ii)} is exactly the T1 regret of the geometric mixture
under the unperturbed model:
$R_0(\hat p_{w^\star})\!-\!R_0(\hat p^\star_0)\!\le\!\tfrac12\sigma_r^{2}V\!+\!O(\sigma_r^{4})$
(\Cref{prop:thm1-step3,lem:excess-risk-log}), where
$V\!=\!\E_{q,c}[\Var_{\hat p_{w^\star}}(\log p_\ctx/p_\theta)]$.
\emph{Terms (i) and (iii)} are TV-perturbation costs of evaluating
the same predictive against two distributions within TV-distance
$\varepsilon$.  For any bounded measurable $f\!:\!\Y\!\to\!\R$
with $\|f\|_\infty\!\le\!B$ and any $\PP, \PP'$ with
$\TV(\PP,\PP')\!\le\!\varepsilon$,
\begin{equation}
|\E_\PP f - \E_{\PP'} f|\;\le\; 2B\,\TV(\PP,\PP')\;\le\;2B\varepsilon
\label{eq:tv-fluct}
\end{equation}
(standard variational identity~\citep[Lemma 2.4]{tsybakov2009nonparametric}).
For log loss, $f(y)\!=\!-\log\hat p(y)\!\in\![0,\log|\Y|]$ on the
geometric mixture under \Cref{app:assumption-bdd-llr}, hence
$\|f\|_\infty\!\le\!\log|\Y|$ and each of (i), (iii) is bounded by
$2\varepsilon\log|\Y|$.  Summing the three terms in \eqref{eq:t11-decomp}
gives the stated bound; the $O(\varepsilon\sigma_r^{2})$ cross-term
captures the fact that the second-order T1 regret itself depends on
$\PP$ through $\Var_{p^\star}$, with the dependence Lipschitz in TV
under \Cref{app:assumption-bdd-llr}.
\end{proof}

\begin{corollary}[Top-$k$ refinement: $\log|\Y|\!\to\!\log k$]
\label{cor:t11-topk}
Restricting to top-$k$ effective support
$\Y_k\!=\!\{y\!:\!\max(p_\theta,p_\ctx)\!\ge\!1/k\}$ replaces
$\log|\Y|$ by $\log k$:
$R_\varepsilon\!-\!R_\varepsilon^\star\!\le\!\tfrac12\sigma_r^{2}V\!+\!4\varepsilon\log k\!+\!O(\varepsilon\sigma_r^{2})$.
For $k\!\sim\!10$--$64$ on conflict cells, the misspec.\ cost is
$\le\!4\varepsilon\log k$ (relative excess
$\sim\!0.4$ at $\varepsilon\!=\!0.1$).
\end{corollary}

\subsubsection{T12: Non-equilibrium tracking bound (transient regret)}
\label{app:t12-non-equilibrium}

\begin{theorem}[Anytime regret of the gen-Bayes trajectory]
\label{thm:t12-non-equilibrium}
For any finite horizon $T\!\ge\!1$ (no requirement of equilibrium
convergence), the cumulative regret of the gen-Bayes trajectory
$\{p_t\}_{t=1}^{T}$ satisfies
\[
\frac{1}{T}\sum_{t=1}^{T}
\bigl[R(p_t)\!-\!R(p^\star)\bigr]
\;\le\;
\frac{C\!\cdot\!\KL(p_0\Vert p^\star)}{\eta T}
\;+\;
\eta\!\cdot\!\Var_{p^\star}\!\bigl[\ell\bigr],
\]
where $\KL(p_0\Vert p^\star)$ is the initial KL gap and the second
term is a constant variance penalty.  Optimizing over $\eta$
gives an $O(\sqrt{\KL/T})$ anytime rate independent of equilibrium
convergence.
\end{theorem}

\begin{proof}[Sketch]
Mirror-descent on $\Lambda_t\!=\!\KL(p_t\|p^\star)$ with drift
$-\eta(R(p_t)\!-\!R(p^\star))\!+\!\eta^2\Var_{p^\star}\ell$;
telescope.
\end{proof}

\begin{lemma}[Gen-Bayes is mirror descent on the natural-parameter chart]
\label{lem:gen-bayes-mirror}
The gen-Bayes update~\eqref{eq:gen-bayes} is exactly mirror descent
with mirror map $\phi(\theta)\!=\!\log\sum_y e^{\theta(y)}$
(log-partition of the categorical exponential family) and step
size $\eta$ on the natural-parameter coordinate
$\theta\!=\!\log p$.  Specifically,
$\theta_{t+1}\!=\!\theta_{t}\!-\!\eta\,\nabla_\theta\E_z[\ell_t(\theta;z)]$
in $\theta$-coordinates is identical to
$p_{t+1}\!\propto\!p_{t}\!\cdot\!\exp(-\eta\E_z\ell_t)$
in probability coordinates.
\end{lemma}

\begin{proof}[Sketch]
$D_\phi$ on $\theta$-coordinates is $\KL$ on $p$-coordinates;
mirror-descent specialises to multiplicative weights via the
exponential change of coordinates~\citep{beck2003mirror}.
\end{proof}

\subsubsection{T13: PAC sample complexity of the plug-in $\hat w(s)$}
\label{app:t13-pac-plugin}

\begin{theorem}[Sample complexity of learning the reliability plug-in]
\label{thm:t13-pac-plugin}
Let $\hat w_n$ be the empirical-risk-minimizer of $\E[r\!\mid\!s]$
over a hypothesis class $\mathcal{H}$ of $L$-Lipschitz functions
with VC-dimension $d_\mathcal{H}$.  With probability $\ge\!1\!-\!\delta$
over $n$ calibration draws,
\[
\E_{q,c}\!\bigl[(\hat w_n(s)\!-\!w^\star(s))^{2}\bigr]
\;\le\;
C_1\,\frac{d_\mathcal{H}\log(n/d_\mathcal{H}\delta)}{n}
\;+\;
\inf_{w\in\mathcal{H}}\E\bigl[(w(s)\!-\!w^\star(s))^{2}\bigr],
\]
and the resulting arbitration regret is
\[
R(\hat p_{\hat w_n})\!-\!R(\hat p_{w^\star})
\;\le\;
V_{\max}\!\cdot\!\E[(\hat w_n\!-\!w^\star)^{2}]
\;+\;O(n^{-1}),
\]
giving the explicit sample-complexity rate
$n\!\ge\!C\,V_{\max}^{2}d_\mathcal{H}\log(1/\delta)/\varepsilon^{2}$
to drive arbitration regret below $\varepsilon$.
\end{theorem}

\begin{proof}[Sketch]
ERM rate~\citep{shalev2014understanding} on VC-bounded $\mathcal{H}$
plus second-order Taylor at $w^\star$ (\Cref{thm:bayes-arbitration}).
\end{proof}

\subsubsection{T14: Adversarially-robust arbitration regret}
\label{app:t14-adversarial}

\begin{theorem}[TV-bounded adversarial signal robustness]
\label{thm:t14-adversarial}
Let $\tilde s$ be an adversarial perturbation of the signal $s$
with $\TV(\PP_s,\PP_{\tilde s})\!\le\!\delta$.  The geometric
mixture under the perturbed signal
$\hat p_{\hat w(\tilde s)}$ has regret bounded by
\[
R(\hat p_{\hat w(\tilde s)})\!-\!R(\hat p^\star)
\;\le\;
\tfrac12\sigma_r^{2}V
\;+\;
2L_w\,\delta\,V_{\max}
\;+\;O(\delta^{2}),
\]
where $L_w$ is the Lipschitz constant of $\hat w(\cdot)$.  In
particular, no adversarial perturbation of the signal can inflate
the regret by more than $O(\delta)$ over the unperturbed Bayes
optimum.  When $\hat w$ is selected from a $1$-Lipschitz class
$\mathcal{H}$, $L_w\!\le\!1$ and the perturbation cost is at most
$2\delta V_{\max}$.
\end{theorem}

\begin{proof}[Sketch]
Lipschitz $\hat w$ gives $|\Delta\hat w|\!\le\!L_w\delta$;
quadratic Taylor on $R(w)$ at $\hat w(s)$ + triangle with
unperturbed bound.
\end{proof}

\subsubsection{T15: Compositional arbitration for $K$-hop chained retrieval}
\label{app:t15-compositional}

\begin{theorem}[Multiplicative composition of reliability weights]
\label{thm:t15-compositional}
For $K$-hop retrieval where context $c_k$ at hop $k$ has reliability
$r_k$ conditional on previous hops $(c_1,\ldots,c_{k-1})$, the joint
arbitration weight on the chain satisfies
\[
w^\star_{1:K}(s_{1:K})
\;=\;
\prod_{k=1}^{K}
w^\star_k(s_k\mid c_{<k})
\quad\text{whenever the per-hop reliability prior factorizes,}
\]
and the geometric-mixture form generalizes to
$\hat p_{w^\star_{1:K}}\!\propto\!p_\theta^{1-\prod_k w^\star_k}\,p_{\ctx_{1:K}}^{\prod_k w^\star_k}$.
\end{theorem}

\begin{proof}[Sketch]
Iterated \Cref{thm:bayes-arbitration} per hop; geometric mixture
closed under natural-parameter composition.
\end{proof}

\begin{remark}[Non-factorisable extension]
Entangled retrieval acquires a copula correction
$w^\star_{1:K}\!=\!\prod_k w^\star_k\cdot\Phi_C(s_{1:K})$, with
$\Phi_C$ estimable~\citep{nelsen2006introduction}; reduces to the
product form under independence.
\end{remark}

\subsubsection{T16: Causal identification of $w^\star$ via doubly-robust IPW}
\label{app:t16-causal}

\begin{theorem}[Causal identification under structural model $q\!\to\!c\!\to\!r\!\to\!Y^\star$]
\label{thm:t16-causal}
Under the structural causal model where $q$ causes $c$, $c$ and
$q$ cause $r$ (treatment), and $r$ together with $(q,c)$ causes
$Y^\star$ (outcome), the arbitration weight $w^\star\!=\!\E[r\!\mid\!q,c]$
is the average treatment effect (ATE) of ``trusting context'' on
$Y^\star$, identified via the doubly-robust
estimator~\citep{robins1994estimation,chernozhukov2018doubleml}:
\[
\widehat{w^\star}_{\mathrm{DR}}
\;=\;
\widehat\mu(s)
\;+\;
\frac{\mathbb{1}[r\!=\!1]\!-\!\widehat\pi(s)}{\widehat\pi(s)}\bigl(r\!-\!\widehat\mu(s)\bigr),
\]
where $\widehat\mu$ is an outcome regression and $\widehat\pi$
is the propensity score.  The DR estimator is $\sqrt n$-consistent
under either-or correctness of $(\widehat\mu,\widehat\pi)$, giving
robustness to misspecification of either nuisance.
\end{theorem}

\begin{proof}[Sketch]
Double-robustness on binary $r$ under no-unmeasured-confounders
$\mid s$; cross-fitted~\citep{chernozhukov2018doubleml}.
\end{proof}

\begin{remark}[Scope of T16]
Requires no-unmeasured-confounders $\mid s$; under
\citet{rosenbaum1983central}-style $\Gamma$-bounded confounding,
$|\widehat w^\star_{\mathrm{DR}}\!-\!w^\star|\!\le\!\Gamma|R\!-\!\widehat\pi|$
is empirically auditable.
\end{remark}

\subsection{Master variational principle and the thermodynamic unification}
\label{app:master-free-energy}

$\mathcal{F}\!=\!\E_p[E]\!-\!H[p]$ is the standard MaxEnt /
ELBO form~\citep{jaynes1957information,wainwright2008graphical};
Genest--Zidek (\citeyear{genest1986characterization}, \S 4)
covers the two-component ($\eta\!=\!0$) case under external
Bayesianity.  Our contribution is the three-component energy
$E_{w,\eta}$ that jointly captures arbitration, tempering, and
post-trace evidence, plus the proof that BayesArbiter's full
rule is the Gibbs minimizer of this single $\mathcal{F}$ ---
\Cref{cor:taxonomy} groups the seven $w^\star$ characterizations
under it.

\begin{theorem}[Master free-energy variational principle]
\label{thm:free-energy-master}
Let
\[
E_{w,\eta}(y;q,c,K) \;:=\;
-(1\!-\!w)\log p_\theta(y\!\mid\!q)
\;-\;w\log p_\ctx(y\!\mid\!q,c)
\;-\;\eta\log p_K(y\!\mid\!q,c,\mathrm{trace})
\]
be the \emph{arbitration-tempering energy} on $\Y$, with
$w\!\in\!\R$ and $\eta\!\in\!\R$ playing the role of inverse
temperatures of the model, context, and post-trace components.
Define the \emph{Helmholtz free energy} of an arbitration policy
$p\!\in\!\Delta^\circ(\Y)$ as
\[
\mathcal{F}[p;w,\eta,q,c,K]
\;:=\;
\E_p[E_{w,\eta}(Y;q,c,K)]\;+\;\E_p[\log p(Y)]
\;=\;
\E_p[E_{w,\eta}]\;-\;H[p].
\]
Then:

\textup{(a) Gibbs form.}  $\mathcal{F}[\,\cdot\,;w,\eta,\cdot]$ is
strictly convex on $\Delta^\circ(\Y)$ and admits the unique minimizer
\begin{equation}
p^\star_{w,\eta}(y)
\;=\;
\frac{p_\theta(y)^{1-w}\,p_\ctx(y)^{w}\,p_K(y)^{\eta}}{Z(w,\eta;q,c,K)},
\qquad
Z(w,\eta;q,c,K)\;=\;\sum_{y'} p_\theta(y')^{1-w}p_\ctx(y')^{w}p_K(y')^{\eta},
\label{eq:master-gibbs}
\end{equation}
with $\min_p\mathcal{F}[p]\!=\!-\log Z(w,\eta;q,c,K)$.

\textup{(b) Bayes-optimal restriction.}  At
$w\!=\!w^\star(s)\!=\!\E[r\!\mid\!s]\!\in\![0,1]$
(\Cref{thm:bayes-arbitration}) and $\eta\!=\!\eta^\star\!\in\![0,1]$
(\Cref{thm:t3-intervention}), $p^\star_{w^\star,\eta^\star}$ is
exactly the BayesArbiter rule.  AdaCAD (variable
$w\!\in\![0,1]$ via JSD, $\eta\!=\!0$), CoCoA (signal-conditioned
$w\!\in\![0,1]$, $\eta\!=\!0$), pure parametric
($w\!=\!0,\eta\!=\!0$), pure context ($w\!=\!1,\eta\!=\!0$), and
post-trace tempering ($w$ fixed, $\eta\!\in\![0,1]$) are
restrictions of $(w,\eta)$ to the convex region $[0,1]^{2}$
where the Gibbs principle is strictly convex.  CAD ($w\!=\!2$)
is the analytic continuation of~\eqref{eq:master-gibbs} to
$w\!>\!1$; the Gibbs form extends as the unique exponential-family
parameterization but the convexity argument no longer applies
(the corresponding ``virtual'' Gibbs measure is normalisable but
not a free-energy minimum on the convex region).

\textup{(c) Bregman / Fisher derivative structure.}
The Bregman divergence with respect to $\mathcal{F}$ in the natural
parameter $\theta\!=\!(\log p_\theta,\log p_\ctx,\log p_K)$ is
\[
D_{\mathcal{F}}(p\,\|\,p^\star_{w,\eta})
\;=\;\KL(p\,\|\,p^\star_{w,\eta})
\;=\;\mathcal{F}[p]\!-\!\mathcal{F}[p^\star_{w,\eta}],
\]
i.e., the KL divergence of any policy from the Gibbs minimizer
\emph{is} its excess free energy.  The Hessian
$\nabla^{2}_\theta\mathcal{F}[p^\star_{w,\eta}]$ is exactly the
Fisher information metric $g_F$ on the exponential family
$\{p^\star_{w,\eta}\}_{(w,\eta)\in\R^{2}}$, recovering
\Cref{thm:info-geometric} as the Hessian-spectral characterization
of $\mathcal{F}$ at its minimum.

\textup{(d) Free-energy decay along gen-Bayes.}  The gen-Bayes
update~\eqref{eq:gen-bayes} is exactly the KL-mirror-descent step
on $\mathcal{F}$:
\[
p_{t+1}\;=\;\arg\min_{p\in\Delta(\Y)}\bigl\{\eta\,\E_p[\E_z\ell_t(Y;z)]\;+\;\KL(p\,\|\,p_t)\bigr\}.
\]
Consequently, $\mathcal{F}[p_t]$ is monotonically non-increasing
for $\eta\!<\!\bar\eta(q,c)$ (\Cref{thm:eta-bayes}), with
descent identity
$\mathcal{F}[p_t]\!-\!\mathcal{F}[p_{t+1}]\!=\!\KL(p_t\,\|\,p_{t+1})\!\cdot\!(1\!+\!O(\eta))$
and asymptotic linear convergence rate
$\rho^\star\!=\!1\!-\!\eta\sigma_{\min}(\nabla^{2}\mathcal{F}[p^\star])/|\Y|\!+\!O(\eta^{2})$
matching \Cref{thm:berk-nash-rate}.

\textup{(e) Critical phase boundary.}  At a hyperbolic Berk--Nash
fixed point $p^\star$, the Hessian
$\nabla^{2}\mathcal{F}[p^\star]\!=\!\mathrm{Cov}_{p^\star}(s_t)$
(exponential-family Fisher identity, equality, by direct
differentiation of $\log Z$ in natural-parameter coordinates).
The critical inverse-temperature at which the smallest eigenvalue
of $\eta\nabla^{2}\mathcal{F}$ crosses unity is therefore
$\eta_c(q,c)\!=\!1/(\sigma_{\min}(\mathrm{Cov}_{p^\star}(s_t))\!\cdot\!\nu)$,
where $\nu$ is the gen-Bayes step normalization fixed by the
mirror-descent identification in (d).  The relationship between
$\sigma_{\min}(\mathrm{Cov}_{p^\star}(s_t))$ and $D^\star(q,c)$
is established by \Cref{lem:score-var-conflict-equality} below
(an equality version of the earlier upper-bound lemma).
Under the $\nu\!=\!|\Y|^{-1}$ convention adopted throughout this
paper~\eqref{eq:gen-bayes}, this gives
$\eta_c\!=\!1/D^\star(q,c)$.  The Hessian zero-crossing is exact
at $\eta_c$ \emph{in the chosen normalization}; under any other
normalization the critical $\eta_c$ scales accordingly but the
existence of a zero-crossing is invariant.  At criticality, the
Hessian's zero-eigenvalue mode dominates and the free-energy
landscape develops a flat direction, giving the polynomial
$t^{-1/2}$ rate of \Cref{thm:phase-transition}.

\textup{(f) Sinkhorn / Schr\"odinger bridge.}  $p^\star_{w,0}$
coincides with the Sinkhorn fixed point at unit temperature with
marginals $(p_\theta, p_\ctx)$
(\Cref{cor:sinkhorn-arbitration}); the geometric mixture is the
$\eta\!=\!0$ projection of the master Gibbs minimizer onto the
arbitration-only sub-manifold.

\textup{(g) Variational characterization of $w^\star$.}
Marginalizing over the prior $\pi(r\!\mid\!s)$,
\[
\E_{r\sim\pi(\cdot\mid s)}\!\bigl[\mathcal{F}[p;r,\eta]\bigr]
\;=\;
\mathcal{F}[p;w^\star(s),\eta]\;+\;\Var_{\pi(r\mid s)}(r)\!\cdot\!V_{\mathrm{post}}(p)/2\;+\;O(\sigma_r^{4}),
\]
which identifies $w^\star\!=\!\E[r\!\mid\!s]$ as the unique
\emph{prior-marginalized} Gibbs minimizer up to the
$\sigma_r^{2}$ Bayes-arbitration regret.  The variance term is
exactly $\E[\sigma_r^{2}(s)]$ from \Cref{thm:t2-info}.
\end{theorem}

\begin{proof}[Sketch]
(a) Lagrangian first-variation gives the Gibbs form; strict
convexity from $-H[p]$.  (b) Direct substitution
(\Cref{cor:special-cases} for $\eta\!=\!0$).  (c)
exponential-family $\KL\!=\!D_{\log Z}$, Hessian $g_F$~\citep[Ch.~3]{amari2000methods}.
(d) Bissiri--Holmes--Walker gen-Bayes is mirror descent with
mirror $\phi\!=\!\log Z$ (\Cref{lem:gen-bayes-mirror}); standard
analysis.  (e) Spectral identity (\Cref{lem:score-var-conflict})
plus $\nu\!=\!|\Y|^{-1}$ yields $\eta_c\!=\!1/D^\star$;
centre-manifold gives $t^{-1/2}$~\citep[Ch.~18]{wiggins2003introduction}.
(f) \citet[Prop.~4.1]{cuturi2013sinkhorn}.  (g) Taylor in $r$
around $w^\star$, marginalize, plus exponential-family identity
$\partial_w\E_{p_w}T\!=\!\Var_{p_w}T$ with $T\!=\!\log p_\ctx/p_\theta$.
\end{proof}

\begin{corollary}[Free-energy taxonomy]
\label{cor:taxonomy}
The seven $w^\star$ characterizations partition into:
\textbf{I} (derived from $\mathcal{F}$ via convex duality):
T1-Bregman (excess $\mathcal{F}$), T8-Fisher (Hessian
$\nabla^2\mathcal{F}\!=\!g_F$), Sinkhorn (marginal projection);
\textbf{II} (same Gibbs form, independent proofs): T1-axiomatic
(Cauchy functional equations), T9-Fano (channel capacity);
\textbf{III} (lower-bound complement): T2-info Pinsker/Le~Cam
floors;
\textbf{IV} (perturbation envelopes of the same $w^\star$):
T11/T13/T14/T16.
\end{corollary}

\begin{remark}[Stat-mech consequences]
Fluctuation--dissipation $\partial_w\E_{p^\star}T\!=\!\Var_{p^\star}T$
($T\!=\!\log p_\ctx/p_\theta$) and Fisher gradient flow
$\dot p\!=\!-g_F^{-1}\nabla_p\mathcal{F}$~\citep[Ch.~5]{amari2000methods}
follow from the master theorem.
\end{remark}

\begin{corollary}[Free-energy monotone descent: a falsifiable trajectory prediction]
\label{cor:F-monotone-descent}
For $\eta\!\le\!\bar\eta(q,c)$ (\Cref{thm:eta-bayes}) and any
gen-Bayes trajectory $\{p_t\}_{t=0}^{K}$, the free-energy sequence
$\{\mathcal{F}[p_t]\}_{t=0}^{K}$ is monotonically non-increasing,
with descent at each step
$\mathcal{F}[p_t]\!-\!\mathcal{F}[p_{t+1}]\!=\!\KL(p_t\,\|\,p_{t+1})\!\cdot\!(1\!+\!O(\eta))$.
The pre-registered prediction is: the empirical
$\widehat{\mathcal{F}}[p_t]$ measured on logged R1-14B,
R1-8B-RLCR, and Mistral-7B traces is monotonically non-increasing
in $t$ for $\eta\!\le\!\bar\eta$, and shows monotone increase
(overshoot) for $\eta\!>\!\bar\eta$.  Falsifying instance: a
single trace where $\widehat{\mathcal{F}}[p_t]$ shows strict
non-monotone behavior in the safe-rate regime would falsify
the master theorem.
\end{corollary}


\paragraph{Joint $(w,\eta)$ decoupling.}
The master theorem decouples in $(w,\eta)$: $\partial_w\mathcal{F}$
fixes $w^\star\!=\!\E[r\!\mid\!s]$ for any $\eta$,
$\partial_\eta\mathcal{F}$ fixes $\eta^\star$ for any $w$.
Empirical correlation $\rho(\widehat w^\star,\widehat\eta^\star)\!=\!-0.06\,(\pm 0.14)$
across $50$ cells, consistent with decoupling.

\subsubsection{Robust min-max arbitration: a consequence requiring the master theorem}
\label{app:robust-minmax-arbitration}

A second falsifiable consequence requiring the master theorem is
the existence of a closed-form \emph{robust min-max arbitration}
policy under prior misspecification.  No individual theorem
(T1--T16) gives this; it follows from the joint variational
structure of $\mathcal{F}$ over $(p, w, \pi)$.

\begin{theorem}[Robust min-max arbitration under prior misspecification]
\label{thm:robust-minmax-arbitration}
Let $\Pi_\varepsilon\!:=\!\{\pi\!:\!\TV(\pi,\pi_0)\!\le\!\varepsilon\}$
be a TV-ball of priors centered at the working prior $\pi_0$.
The robust min-max arbitration problem
\[
(p^\sharp_\varepsilon, w^\sharp_\varepsilon)
\;:=\;
\arg\min_{p\in\Delta(\Y),\,w\in[0,1]}
\max_{\pi\in\Pi_\varepsilon}
\E_{r\sim\pi}\!\bigl[\mathcal{F}[p; r, \eta]\bigr]
\]
admits the closed-form solution
\[
p^\sharp_\varepsilon
\;\propto\;
p_\theta^{1-w^\sharp_\varepsilon}\,p_\ctx^{w^\sharp_\varepsilon}\,p_K^{\eta},
\qquad
w^\sharp_\varepsilon
\;=\;
\E_{\pi_0}[r]\;+\;C_{\mathrm{rob}}\!\cdot\!\varepsilon\;+\;O(\varepsilon^2),
\]
with explicit leading constant
\begin{equation}
C_{\mathrm{rob}}
\;=\;
\frac{\mathrm{sign}(\partial_w R|_{w=\E_{\pi_0}[r]})\,\Vert \log(p_\ctx/p_\theta)\Vert_\infty}{\partial^{2}_{ww}R|_{w=\E_{\pi_0}[r]}},
\quad
\partial^{2}_{ww}R\!=\!2\E\bigl[\Var_{p^\star_{w}}(\bm{1}\{Y\!=\!y^\star\})(\log p_\ctx/p_\theta)^{2}\bigr]
\label{eq:robust-leading-constant}
\end{equation}
where the Hessian $\partial^{2}_{ww}R$ is the Brier curvature of
the risk in $w$ at the unperturbed optimum, computed via the
fluctuation--dissipation identity
$\partial_w\E[T]\!=\!\Var(T)$ in (g) of \Cref{thm:free-energy-master}.
For unbounded $\log(p_\ctx/p_\theta)$ the bound becomes
$|w^\sharp_\varepsilon\!-\!\E_{\pi_0}[r]|\!\le\!\sqrt{2\varepsilon/\partial^{2}_{ww}R}$
(the $\sqrt\varepsilon$ scaling is sharp under sub-Gaussian
log-ratios; an explicit two-state example saturates it).
\end{theorem}

\begin{proof}[Sketch]
$\mathcal{F}$ is convex in $p$, linear in $\pi$; Sion swap, TV
variational identity~\citep{villani2009optimal} gives
linear-in-$\varepsilon$ gradient perturbation, second-order Taylor
on $R$ closes \eqref{eq:robust-leading-constant}.
$\partial^2_{ww}R\!>\!0$ by Berk--Nash non-degeneracy.
\end{proof}

\begin{remark}[Empirical estimability]
\label{rem:robust-constant-empirical}
$\partial^2_{ww}R$ is twice the empirical
$\Var_{p^\star_{\hat w}}(\bm{1}\{Y\!=\!y^\star\})$ weighted by the
squared log-ratio; per-cell estimates are in the release
artefact, falsifiable at $\varepsilon\!=\!0.05$ prior-noise
injection.
\end{remark}

\subsection{Safe-rate corollary and Berk--Nash convergence detail}
\label{app:safe-rate-detail}

\paragraph{Safe rate.}
\citet[Thm.~1.6]{grunwald2017}: $\bar\eta\!\le\!c_1\varepsilon^*(D^\star)/D^\star$
with $\varepsilon^*$ solving the bracketing-entropy fixed point.
Polynomial bracketing $\log N_{[]}(\varepsilon)\!\le\!C\varepsilon^{-\rho}$
($\rho\!\in\!(0,2)$) gives $\bar\eta\!\le\!C'/D^\star$.  Finite
$|\Y|\!=\!m$: $\log N_{[]}(\varepsilon,\Delta(\Y))\!\le\!(m\!-\!1)\log(3/\varepsilon)$,
yielding prefactor $c_{|\Y|,\pi}\!=\!c_0\Gamma(m,\pi)$.

\paragraph{Conflict raises $D^\star$.}
Identification assumes $D^\star_{\mathrm{conflict}}\!>\!D^\star_{\mathrm{no-conflict}}$
on the slices of interest; the safe-rate bound then forces the
ordering on $\bar\eta$.  Empirical scale check (R1-14B/ConflictBank,
$\delta\!\approx\!2.31$ nats): predicted ratio $\sim\!0.34$,
realised $\sim\!0.35$.

\paragraph{Berk--Nash convergence.}
Best-response $T(p)\!=\!\arg\max_{q_\xi\in\mathcal{M}}\E_{z\sim p}\log q_\xi(z)$
is continuous (\citealp{esponda2016}, Asm.~1), so Brouwer gives a
fixed point.  Endogenous-evidence trajectory converges almost
surely~\citep[Thm.~1]{fudenberg2021}.  Under misspecification the
fixed point is the KL projection, not the truth — projection
distance $D^\star$ measures the gap.

\subsection{Derivation of $\eta$-tempered decoding}
\label{app:eta-tempering-derivation}

If the post-trace answer distribution is a Gibbs posterior at
effective rate $\eta_{\mathrm{eff}}\!>\!\bar\eta$, shrinking back
to $\eta'\!\le\!\bar\eta$ is the exponentiation
$p_\eta\!\propto\!p^{\eta'/\eta_{\mathrm{eff}}}$.  Brier-optimised
$\eta$ on a held-out split is~\citet{grunwald2017}'s safe-Bayes
adapted to inference time.

\subsection{Algorithms}
\label{app:algorithms}

\subsubsection{Plug-in Bayes proxy (Theorem~1)}

\begin{algorithm}[H]
\small
\caption{\textsc{BayesArbiter} plug-in.  Per-benchmark training fits one
reliability model; test-time inference is one geometric mixture.}
\begin{algorithmic}
\Require Calibration set $\D_{\mathrm{cal}}$ with $(q_i,c_i,y_i^\star)$,
closed-book distributions $p_\theta(\cdot\mid q_i)$, context-conditioned
distributions $p_{\ctx}(\cdot\mid q_i,c_i)$
\Ensure Prediction $\hat y$ and arbitration distribution $\hat p(\cdot\mid q,c)$
for a new example $(q,c)$
\Statex \textit{Calibration stage}
\State $\D_{\mathrm{conf}}\gets \{i:y_{\ctx,i}\neq y_{\param,i}\}$
\ForAll{$i\in\D_{\mathrm{conf}}$}
  \State compute features $s_i\gets(\text{BM25},\text{popularity},\text{sem.\ entropy},\text{ctype})$
  \State label $r_i\gets \one{y_i^\star=y_{\ctx,i}}$
\EndFor
\State fit $\hat w(s)=\sigma(\hat\beta^\top s)$ by L2 logistic regression on
$\{(s_i,r_i)\}_{i\in\D_{\mathrm{conf}}}$
\Statex \textit{Inference stage}
\State compute $s(q,c)$ and $\hat w\gets \sigma(\hat\beta^\top s(q,c))$
\ForAll{$y\in\Y$}
  \State $\log \tilde p(y)\gets (1-\hat w)\log p_\theta(y\mid q)+\hat w\log p_{\ctx}(y\mid q,c)$
\EndFor
\State $\hat p(\cdot\mid q,c)\gets \softmax(\log \tilde p)$;
$\hat y\gets \arg\max_{y\in\Y}\hat p(y\mid q,c)$
\State \Return $(\hat y,\hat p)$
\end{algorithmic}
\end{algorithm}

\subsubsection{$\eta$-tempered decoding (Theorem~3)}

\begin{algorithm}[H]
\small
\caption{$\eta$-tempered post-trace decoding.  $\eta^\star$ selected
per-(model, benchmark, condition) on a held-out split, with do-no-harm
fallback.}
\begin{algorithmic}
\Require Calibration split $\D_{\mathrm{cal}}$, test example $(q,c)$,
trace budget $K$, candidate grid $\Gamma\subset[0,1]$
\Ensure Tempered post-trace distribution $p_{\eta^\star}$
\Statex \textit{Generate post-trace distributions}
\ForAll{$(q_i,c_i)\in\D_{\mathrm{cal}}\cup\{(q,c)\}$}
  \State sample CoT $z_{1:K}^{(i)}$ with the base model;
  compute $p_i(\cdot)\!=\!p(\cdot\mid q_i,c_i,z_{1:K}^{(i)})$
\EndFor
\Statex \textit{Held-out rate selection}
\ForAll{$\eta\in\Gamma$}
  \State $p_{i,\eta}\gets p_i^\eta/\sum_{y'} p_i(y')^\eta$ for each $i\in\D_{\mathrm{cal}}$
  \State record held-out Brier and accuracy at $\eta$
\EndFor
\State $\eta^\star\gets \arg\min_{\eta\in\Gamma}\mathrm{Brier}(\eta)$
\State (do-no-harm) replace $\eta^\star$ by the largest $\eta\in\Gamma$ with held-out
accuracy $\geq$ $\eta\!=\!1$ baseline if any tie
\Statex \textit{Decode the test example}
\State return $p_{\eta^\star}(\cdot\mid q,c,z_{1:K})\propto p(\cdot\mid q,c,z_{1:K})^{\eta^\star}$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
\small
\caption{Confidence-head pilot.  A lightweight confidence regressor trained
on frozen activations as a proxy for full RLCR-style retraining.}
\begin{algorithmic}
\Require Examples with answer-token hidden states $h_i$ and binary correctness
labels $c_i\in\{0,1\}$
\Ensure Confidence head $g_\phi(h)$
\Statex \textit{Feature extraction}
\ForAll{example $i$}

\State \Return $g_\phi$
\end{algorithmic}
\end{algorithm}

\subsubsection{Equivalence of Brier and log-loss minimizers in the geometric family}

\begin{proposition}[Brier and log-loss agree on the Bayes optimum]
\label{prop:brier-log-equivalence}
Let $p^\star$ denote the Bayes-optimal predictive distribution under log
loss (i.e., the posterior predictive).  Then $p^\star$ also minimizes the
expected Brier score.  Equivalently, on the log-linear family the
Brier and log-loss objectives agree whenever the family contains
$p^\star$.  In the approximate log-linear setting used in the body, both
objectives are optimized at nearby weights when the variance-correction
term is small.
\end{proposition}

\begin{proof}[Sketch]
Both proper, so both minimised at the posterior predictive
(\Cref{prop:thm1-step1}); when the family contains $p^\star$
the minimisers agree, otherwise they remain close as the
variance-correction shrinks (\Cref{prop:thm1-step3}).
\end{proof}

\subsubsection{Multi-source generalization}

\begin{proposition}[Multi-source arbitration]
\label{prop:multi-source}
Suppose there are $K$ candidate sources with answer distributions
$p_1,\ldots,p_K$ and a simplex-valued reliability variable
$r\in\Delta^{K-1}$.  Under the multi-source generalization of
\Cref{app:assumption-loglinear},
$p(y\mid q,c,r)\propto\prod_k p_k(y)^{r_k}$.  The Bayes-optimal
predictive distribution admits the same variational projection as in the
binary case, with geometric mean
$p^\star\!\propto\!\prod_k p_k^{w^\star_k}$ with
$w^\star\approx\E[r\mid q,c]\in\Delta^{K-1}$ when the posterior over $r$
is concentrated.  The plug-in $\hat w$ from a
multinomial logistic regression on $s(q,c)$ achieves the same
$\sqrt n$-consistency rate as the binary case under standard
multinomial-regression regularity assumptions.
\end{proposition}

\begin{proof}[Sketch]
Multi-marginal KL $\sum_k w_k\KL(p\|p_k)$ is uniquely minimised at
$p_w\!\propto\!\prod_k p_k^{w_k}$; concentrated $r$ + Taylor gives
$p^\star\!=\!p_{w^\star}\!+\!O(\mathrm{tr}\,\Var[r])$; multinomial
M-estimation gives $\sqrt n$-consistency.
\end{proof}

\subsubsection{Moment-form restatement of the plug-in inequality}

\begin{proposition}[Moment-form excess-risk bound]
\label{prop:lp-excess}
The quadratic log-loss bound of \Cref{lem:excess-risk-log} can be
rewritten in terms of any moment control that upper bounds
$\E[(\hat w-w^\star)^2]$.  In particular, if
$\E[(\hat w-w^\star)^2]\le \delta_n$, then
$R_{\log}(\hat p_{\hat w})-R_{\log}(\hat p_{w^\star})\le 2M^2\delta_n$.
\end{proposition}

\begin{proof}
Immediate from \Cref{lem:excess-risk-log}.
\end{proof}

\subsection{Variants of $\eta$-tempering}
\label{app:eta-variants}

(A) per-token $\eta$ ($-0.05$ Brier vs.\ single-token);
(B) confidence-conditional $\eta(c)\!=\!1\!-\!0.9\one{\max_y p\!\ge\!0.95}$
($+0.02$ Brier on worst slice);
(C) trace-length $\eta(K)\!=\!1\!-\!0.5K/K_{\max}$
($+0.01$ Brier);
(D) KL-projection $\eta$ ($\sim\!0.5$, not competitive with
$\eta^\star\!=\!0$ Brier-optimal).  Sample-split Brier-optimal
remains the default.


\subsection{Supplementary derivation, $\eta$ from log-odds}
\label{app:eta-from-logodds}

For binary $\Y\!=\!\{0,1\}$, post-trace log-odds
$\ell\!=\!\log[p(1)/p(0)]$ shrink to $\ell_\eta\!=\!\eta\ell$
(one multiply).  For multinomial $|\Y|\!>\!2$:
$\eta\log p(y)$ + softmax — the $\eta$-Gibbs correction at the readout
(\Cref{app:eta-tempering-derivation}); $\eta\!=\!0$ gives uniform,
$\eta\!=\!1$ leaves $p$ unchanged.  Shrinking toward $p_\theta$ (rather
than uniform) is empirically slightly worse on the worst slice
($-0.50$ vs.\ $-0.40$ Brier); uniform is the safer default.

\subsection{Bootstrap, e-value, and glossary}
\label{app:bootstrap-evalue}

\textbf{Inference procedures.}  Paired bootstrap CIs from $10{,}000$
benchmark$\times$model resamples ($2.5/97.5$ quantiles); exact
one-sided sign test is the binomial-tail on series-level Bayes wins.
Fixed-$\lambda$ e-value~\citep{shafer2021testing}:
$E(\lambda)=\prod_{i=1}^{25}[1+\lambda(X_i^{\mathrm{Bayes}}-X_i^{\mathrm{heur}})/c]$
with $c$ enforcing $|M_i-1|\!\le\!1$ and $\lambda$ from LOO-CV; the
reported $E\!=\!2806$ ($p\!\sim\!10^{-4}$ via $1/E$) is anytime-valid
and robust to optional stopping~\citep{ramdas2023game}.

\textbf{Glossary.}  Arbitration weight $w$: weight on $p_\ctx$ in the
geometric mixture.  Reliability $r$: $\PP(y^\star\!=\!y_\ctx\mid y_\ctx\!\neq\!y_\param)$.
Posterior reliability $w^\star\!=\!\E[r\mid q,c]$; plug-in $\hat w$:
logistic regression on $(\mathrm{BM25},\mathrm{pop.},\mathrm{sem.\ ent.},\mathrm{ctype})$.
Safe rate $\bar\eta$: Gr\"unwald--van~Ommen rate; $\eta_{\mathrm{eff}}$:
empirical rate of \eqref{eq:gen-bayes}.  Saturation: $\eta\!>\!\bar\eta$;
peak-then-recover: $\eta\!\lesssim\!\bar\eta$.
Gap $G(K)\!=\!|\overline{\mathrm{conf}}(K)\!-\!\mathrm{acc}(K)|$;
$\eta$-tempering: post-trace exponentiation by $\eta\!\in\![0,1]$.

\textbf{Method choices.}
\emph{Regret over accuracy:} accuracy on conflict is bounded above by

$0.001$ regret (zero within SE).  Logistic is auditable per coefficient.

\subsection{Pre-registered held-out test}
\label{app:preregistered-heldout}

We pre-registered three predictions before running on two held-out
families that were not used to select hyperparameters or signal
weights.  The held-out families are ClashEval and RAMDocs.

\textbf{P1 (T1, supported).} On both held-out families, the
Bayes-proxy geometric mixture beats the heuristic adaptive
baseline, CoCoA, Astute RAG, and Self-RAG at the point estimate,
with the same monotone ordering in residual gap as on the
in-package spotlight matrix.

\textbf{P2 (T2, supported).} The boundary policies (always-context,
always-parametric) remain catastrophically worse than every
adaptive policy, by the same orders-of-magnitude margin as in
\Cref{tab:headline-regret}.

\textbf{P3 (T3, not supported).} The strongest directional
prediction of T3 (saturation specifically on conflict cells with
$\bar\eta$ shrinking with $\KL(p_\theta\|p_{\ctx})$) does not
replicate on the held-out ClashEval and RAMDocs cells in the form
we pre-registered.  The benchmark-dependence of saturation on
ConflictBank and WikiContradict (\Cref{tab:theorem3-matrix-full}) does
not transfer cleanly to ClashEval or RAMDocs, where the
no-conflict baselines themselves degrade enough at long $K$ that
the conflict-vs-no-conflict separation is muddied.  The
inference-time intervention ($\eta$-tempering) still helps on the
held-out cells in the same direction as in
\Cref{tab:eta-per-slice}, but the saturation-classification claim
is the part that does not survive the held-out test as
pre-registered.  We read this as a genuine limit on T3's
generality, not as a refutation; the safe-rate mechanism is still
visible in the $\eta^\star$ ranking
(\Cref{app:reviewer-qa}, Q3) but the binary saturate/recover
classifier is benchmark-dependent in a way the pre-registration
did not anticipate.

\subsection{Adversarial / poisoned context}
\label{app:adversarial-context}

We construct \emph{poisoned} cells by injecting an adversarial passage
into the retrieved context that contradicts the gold answer.  The

\paragraph{Bayes-proxy snippet (vectorized over the calibration set).}
\begin{verbatim}
import numpy as np
from sklearn.linear_model import LogisticRegression

# (1) calibration features and labels
S = np.array([retrieval_score, popularity, semantic_entropy,
              contradiction_type]).T            # shape (n, 4)
r_obs = (y_star == y_ctx).astype(int)            # observed reliability
mask  = (y_ctx != y_param)                       # conflict subset

# (2) fit plug-in w_hat
clf = LogisticRegression(C=1.0).fit(S[mask], r_obs[mask])

# (3) inference-time geometric mixture
def bayes_proxy_predict(p_theta, p_ctx, s):
    w_hat = clf.predict_proba(s.reshape(1, -1))[0, 1]
    log_p = (1 - w_hat) * np.log(p_theta) + w_hat * np.log(p_ctx)
    return np.exp(log_p - np.logaddexp.reduce(log_p))
\end{verbatim}

\paragraph{$\eta$-tempering snippet (with do-no-harm fallback).}
\begin{verbatim}
def eta_temper(p_post, eta):
    log_p = eta * np.log(p_post + 1e-12)
    return np.exp(log_p - np.logaddexp.reduce(log_p))

def select_eta(p_calib, y_calib,
               grid=np.linspace(0, 1, 11), do_no_harm=True):
    base_acc = np.mean(np.argmax(p_calib, -1) == y_calib)
    best_eta, best_brier = 1.0, np.inf
    for eta in grid:
        p_t   = np.array([eta_temper(p, eta) for p in p_calib])
        brier = np.mean(np.sum((p_t - one_hot(y_calib))**2, -1))
        acc   = np.mean(np.argmax(p_t, -1) == y_calib)
        if (not do_no_harm or acc >= base_acc) and brier < best_brier:
            best_eta, best_brier = eta, brier
    return best_eta
\end{verbatim}

\subsubsection{T1--T3 assumptions on closed-weight production models}
\label{app:closed-model-assumptions}

All load-bearing T1--T3 assumptions reduce to task-level
conditions or output-type conditions, both
weight-access-independent: latent $r$ (task), log-linear softmax
over discrete $\Y_q$ (output head), $y^\star\!\perp\!c\mid(q,r)$
(task), two opposite sub-families (task), discriminative $s(q,c)$
(observable from text I/O), bounded $\ell$ + Lipschitz sampler

(App.~\ref{app:closed-model-deployment}).  Empirically:
GPT-4o-mini / Claude-3.5-Haiku / Gemini-1.5-Flash all show Bayes

Bayes\,$>$\,AdaCAD\,$>$\,CoCoA\,$>$\,Astute\,$>$\,Self-RAG ordering
preserved.  Non-applicability: APIs that expose only a greedy
sample with no logprobs and no structured-output support; MoE with
non-shared tokenizers across experts.

\subsubsection{Closed-model deployment recipe}
\label{app:closed-model-deployment}

The geometric mixture is evaluable from API surfaces in three
steps: (1)~restrict to the discrete candidate set $\Y_q$
($|\Y_q|\!\le\!20$ on every spotlight benchmark); (2)~extract
per-candidate scalars via top-$k$ logprobs (OpenAI, Gemini),
elicited confidence with $K\!=\!8$ resampled draws (Anthropic),
or structured-output sampling at $K\!=\!16$ (universal fallback);
(3)~fit $\hat w$ on text-only signals (BM25, popularity, semantic
entropy~\citep{kuhn2023semantic} from $K\!=\!16$ samples).  Closed
and open numbers are proxies, not directly comparable on the same
scale; we do not mix them in the powered $n\!=\!210$ head-to-head.
Per-query overhead $\sim\!2.3\!\times$ latency / $\sim\!2.1\!\times$
cost.  Non-applicable: APIs returning only a single sample with no
logprobs, no structured output, no resampling.

\section{Limitations, broader impact, threats, and reviewer Q\&A}
\label{app:letter-M}

\subsection{Limitations}
\label{app:limitations}

\textbf{Theory.}  T1 assumes log-linear conditional answer model
(\Cref{app:assumption-loglinear}) and $y^\star\!\perp\!c\mid (q,r)$;
gated mixtures left to future work.  T2's $\Omega(1)$ lower-bound
constant is loose on natural benchmarks.  T3 is most fragile:
Qwen-32B/ConflictBank is the one cell where the
$\widehat\eta_{\mathrm{eff}}/\bar\eta$ ordering misses an absolute
confidence floor not captured by the trajectory-level inequality.
\textbf{Empirics.}  MQuAKE is policy-probability proxy, not free-form
multi-hop; retrieval demo is BM25 over WikiContradict's own corpus
(no open-Wikipedia production retrieval); confidence-head pilot is
$180/60/60$ at $20$ epochs; English-only.


\subsection{Broader impact}
\label{app:broader-impact}

Positive: reliability-aware arbitration reduces the headline failure
modes of fixed-$w$ decoders (parroting outdated/poisoned passages;
refusing to update on new evidence) in conflict-heavy verticals
(medical labels, regulatory texts, contested literature).
Negative: any rule that decides \emph{how much to trust retrieval}
also decides \emph{how much to trust specific sources}; popularity
and retrieval-score signals carry their own biases and may
underweight context for rare entities or marginalized perspectives,
so source-trust-sensitive deployments need an explicit audit step.
\Cref{thm:eta-bayes}'s implication---RLVR reasoning models can be
\emph{more} confidently wrong than chat models on knowledge-conflict
slices---argues for a confidence-only calibration step
(\Cref{tab:eta-per-slice}) and the two-regime trajectory diagnostic
(\Cref{cor:two-regime}) in safety-critical reasoning deployments.
No model release; artifacts are evaluation scripts and per-row CSVs.


\subsection{Ethics, dual-use, and responsible release}
\label{app:ethics-dual-use}

\textbf{Dual-use.}  An arbitration rule that picks $w$ can also be
inverted to suppress inconvenient context; we recommend exposing
per-query $\hat w(c)$ for operator-side auditability.
\textbf{Signal bias.}  Popularity favours rich parametric coverage;
per-domain audits should check calibration on protected attributes.
\textbf{Do-no-harm.}  $\eta$-tempering only deflates overconfidence;
it never converts a correct answer to wrong.
\textbf{Reproducibility.}  Public benchmarks, open-weight models,
no human subjects, no PII, no model release.


\subsection{Threats to validity}
\label{app:threats}

\textbf{Construct.}  Latent $r\!=\!\PP(y^\star\!=\!y_\ctx\mid y_\ctx\!\neq\!y_\param)$
requires both closed-book and context-conditioned answers to commit;
we restrict to $\max_y p_\theta(y)\!\ge\!0.5$ and $\max_y p_\ctx(y)\!\ge\!0.5$.
\textbf{Internal.}  Signals are correlated (retrieval score with
popularity, entropy with both); L2-regularised logistic plug-in
handles this and saturates by $n\!=\!512$ (\Cref{tab:sample-efficiency}).
\textbf{External.}  $13$ models, $10$ benchmarks, English-only;
no conversational, multilingual, or vertical-specific settings.
\textbf{Statistical.}  $25$ benchmark$\times$model series is small
for a sign test; we report bootstrap CIs, sign tests, and the
e-value of \citet{shafer2021testing} for transparency.
